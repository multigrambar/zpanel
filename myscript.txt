-- Persist on teleport
-- Re-injects the script automatically when you hop to a different
-- server (Roblox kills LocalScripts on teleport; queue_on_teleport
-- is the only way to survive it).
--
-- Works on Potassium (uses bare `queue_on_teleport` global per their
-- docs), Synapse (`syn.queue_on_teleport`), Fluxus, KRNL, etc.
--
-- USAGE: paste your loadstring URL into _G.LT2_SCRIPT_URL before
-- running the script, e.g.
--   _G.LT2_SCRIPT_URL = "https://raw.githubusercontent.com/.../myscript.txt"
--   loadstring(game:HttpGet(_G.LT2_SCRIPT_URL))()
-- We then pass a wrapped loadstring of that URL to whichever
-- queue_on_teleport function the executor exposes.
------------------------------------------------------------------
-- Default URL — overridable via _G.LT2_SCRIPT_URL before running
if not _G.LT2_SCRIPT_URL then
    _G.LT2_SCRIPT_URL = "https://raw.githubusercontent.com/multigrambar/zpanel/refs/heads/main/myscript.txt"
end

-- queue_on_teleport detector — Potassium uses the bare global per
-- their docs; Synapse/Fluxus use namespaced ones as backup paths
local function getQOT()
    return queue_on_teleport
        or (syn and syn.queue_on_teleport)
        or (fluxus and fluxus.queue_on_teleport)
        or (getgenv and getgenv().queue_on_teleport)
end

-- Build the payload — includes any pending-find state so a chain of
-- hops keeps the find alive. extraCode is prepended raw, before the
-- loadstring, so things like `_G.LT2_PendingFind = "LoneCave"` happen
-- BEFORE the script reloads.
local function buildPayload(extraCode)
    return (extraCode or "")
        .. ([[loadstring(game:HttpGet("%s"))()]]):format(_G.LT2_SCRIPT_URL)
end

local function fireQueue(extraCode, reason)
    local qot = getQOT()
    if not qot then
        warn("[ZPanel] No queue_on_teleport available — script won't reload")
        return false
    end
    local payload = buildPayload(extraCode)
    local ok, err = pcall(function() qot(payload) end)
    if ok then
        warn("[ZPanel] Queued re-inject (" .. (reason or "manual") .. ")")
    else
        warn("[ZPanel] queue_on_teleport failed: " .. tostring(err))
    end
    return ok
end

-- IMMEDIATE queue at startup (covers manual Roblox menu teleports)
fireQueue(nil, "startup")

-- Hook OnTeleport — the canonical pattern from executor docs. Fires
-- the moment the player initiates ANY teleport, so we re-queue even
-- if our startup call got consumed by an earlier teleport.
do
    local ok = pcall(function()
        LocalPlayer.OnTeleport:Connect(function(state)
            if state == Enum.TeleportState.Started
            or state == Enum.TeleportState.InProgress
            or state == Enum.TeleportState.WaitingForServer then
                -- Preserve any pending state for the next server
                local extra = ""
                if _G.LT2_PendingFind then
                    extra = string.format("_G.LT2_PendingFind = %q\n",
                        _G.LT2_PendingFind)
                end
                fireQueue(extra, "OnTeleport." .. tostring(state))
            end
        end)
    end)
    if not ok then
        warn("[ZPanel] OnTeleport hook failed — manual teleports may not reload")
    end
end

------------------------------------------------------------------
-- Server-hop helper + pending-find auto-retry.
-- Used by Wood Finder when a rare tree (LoneCave/Spooky) isn't in
-- the current server. We pass _G.LT2_PendingFind through the
-- queue_on_teleport payload so the find auto-resumes on the next
-- server. After the script reloads on the new server, a delayed
-- task watches for _G.LT2_PendingFind + re-fires the find.
------------------------------------------------------------------
local TeleportService = game:GetService("TeleportService")
local HttpService     = game:GetService("HttpService")
local LT2_PLACE_ID    = 13822889  -- Lumber Tycoon 2

-- Fetch public servers from Roblox API; return shuffled jobIds
local function getOtherServers()
    local url = "https://games.roblox.com/v1/games/" .. LT2_PLACE_ID
              .. "/servers/Public?sortOrder=Asc&limit=100"
    local fetcher = (syn and syn.request and function(u)
        return syn.request({Url = u, Method = "GET"}).Body
    end) or (http_request and function(u)
        return http_request({Url = u, Method = "GET"}).Body
    end) or function(u) return game:HttpGet(u) end

    local ok, body = pcall(fetcher, url)
    if not ok or type(body) ~= "string" then return {} end
    local pok, data = pcall(function() return HttpService:JSONDecode(body) end)
    if not pok or type(data) ~= "table" then return {} end

    local list = {}
    for _, s in ipairs(data.data or {}) do
        if s.playing and s.maxPlayers
        and s.playing < s.maxPlayers and s.id ~= game.JobId then
            table.insert(list, s.id)
        end
    end
    -- Fisher-Yates shuffle
    for i = #list, 2, -1 do
        local j = math.random(1, i)
        list[i], list[j] = list[j], list[i]
    end
    return list
end

-- Hop to a random other LT2 server. Pre-queues the script with
-- extraQueueCode prepended so state (like pending-find) survives.
-- Tries up to 5 different jobIds in case one rejects.
local function serverHop(extraQueueCode)
    local servers = getOtherServers()
    if #servers == 0 then return false, "API returned no servers" end
    -- Pre-fire the queue NOW (before teleport starts) so it's set
    -- regardless of whether OnTeleport fires fast enough
    fireQueue(extraQueueCode, "pre-hop")
    -- Try a few servers in case one is full / private / rejected
    for i = 1, math.min(5, #servers) do
        local ok, err = pcall(function()
            TeleportService:TeleportToPlaceInstance(LT2_PLACE_ID, servers[i], LocalPlayer)
        end)
        if ok then return true end
        warn("[ZPanel] Hop attempt " .. i .. " failed: " .. tostring(err))
        task.wait(0.3)
    end
    return false, "All hop attempts rejected"
end

-- Pending-find handler: runs after the panel boots if a previous
-- server-hop left _G.LT2_PendingFind set. Re-fires the find via
-- _G.FindTreeAndHop after a short delay so PlayerGui is ready.
if _G.LT2_PendingFind then
    local target = _G.LT2_PendingFind
    _G.LT2_PendingFind = nil
    task.spawn(function()
        task.wait(5)  -- let panel + workspace populate
        if _G.FindTreeAndHop then _G.FindTreeAndHop(target) end
    end)
end

local themed = {}
local function themeify(instance, property, themeKey)
    table.insert(themed, {instance, property, themeKey})
    instance[property] = Theme[themeKey]
end
local function retagTheme(instance, property, themeKey)
    for _, e in ipairs(themed) do
        if e[1] == instance and e[2] == property then e[3] = themeKey; return end
    end
end

local function tween(o, t, p, style, dir)
    TweenService:Create(o, TweenInfo.new(
        t,
        style or Enum.EasingStyle.Quad,
        dir   or Enum.EasingDirection.Out
    ), p):Play()
end
-- Subtle 3px rounding across the whole panel — modern but not pillowy.
-- Ignores the passed-in radius so the whole UI is consistent.
local function corner(p, _r)
    local c = Instance.new("UICorner"); c.CornerRadius = UDim.new(0, 3); c.Parent = p; return c
end
local function stroke(p, color, t)
    local s = Instance.new("UIStroke"); s.Color = color; s.Thickness = t or 1
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border; s.Parent = p; return s
end
local function pad(p, n)
    local pa = Instance.new("UIPadding")
    pa.PaddingTop, pa.PaddingBottom = UDim.new(0, n), UDim.new(0, n)
    pa.PaddingLeft, pa.PaddingRight = UDim.new(0, n), UDim.new(0, n)
    pa.Parent = p; return pa
end
-- Gradient helper kept as a no-op (caller-friendly) so old call sites
-- don't crash. We want a flat look — no gradients anywhere.
local function gradient(_parent, _c1, _c2, _rot)
    return nil
end
-- Glow helper kept as a no-op stub — returns a fully-invisible ImageLabel
-- so callers that set .ImageColor3 / .ImageTransparency / .Size or feed
-- it to TweenService don't error. We want a flat look, no halos.
local function glow(parent, _color, _intensity)
    local g = Instance.new("ImageLabel")
    g.Name = "GlowStub"
    g.Size = UDim2.new()
    g.Visible = false
    g.BackgroundTransparency = 1
    g.ImageTransparency      = 1
    g.Parent = parent
    return g
end

-- Fly
------------------------------------------------------------------
local Fly = { enabled=false, speed=60, maxTilt=35, smoothness=6, parts=nil, velocity=Vector3.zero }

local function flyCleanup()
    if Fly.parts then
        pcall(function() Fly.parts.lv:Destroy() end)
        pcall(function() Fly.parts.ao:Destroy() end)
        pcall(function() Fly.parts.att:Destroy() end)
        Fly.parts = nil
    end
    local char = LocalPlayer.Character
    if char then local h=char:FindFirstChildOfClass("Humanoid"); if h then h.PlatformStand=false end end
    Fly.velocity = Vector3.zero
end
local function flySetup()
    flyCleanup(); if not Fly.enabled then return end
    local char=LocalPlayer.Character; if not char then return end
    local hrp=char:FindFirstChild("HumanoidRootPart")
    local hum=char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then return end
    hum.PlatformStand=true
    local att=Instance.new("Attachment"); att.Name="FlyAttachment"; att.Parent=hrp
    local lv=Instance.new("LinearVelocity")
    lv.Attachment0=att; lv.MaxForce=math.huge; lv.VectorVelocity=Vector3.zero
    lv.RelativeTo=Enum.ActuatorRelativeTo.World; lv.Parent=hrp
    local ao=Instance.new("AlignOrientation")
    ao.Attachment0=att; ao.Mode=Enum.OrientationAlignmentMode.OneAttachment
    ao.MaxTorque=math.huge; ao.Responsiveness=35; ao.Parent=hrp
    Fly.parts={att=att,lv=lv,ao=ao}
end
local function flySetEnabled(v) Fly.enabled=v; flySetup() end

bind(RunService.RenderStepped:Connect(function(dt)
    if not Fly.enabled or not Fly.parts then return end
    local cam=workspace.CurrentCamera; if not cam then return end
    local cf=cam.CFrame; local move=Vector3.zero
    if UserInputService:IsKeyDown(Enum.KeyCode.W)           then move+=cf.LookVector  end
    if UserInputService:IsKeyDown(Enum.KeyCode.S)           then move-=cf.LookVector  end
    if UserInputService:IsKeyDown(Enum.KeyCode.D)           then move+=cf.RightVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.A)           then move-=cf.RightVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.Space)       then move+=Vector3.yAxis  end
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then move-=Vector3.yAxis  end
    if move.Magnitude>0.001 then move=move.Unit end
    Fly.velocity=Fly.velocity:Lerp(move*Fly.speed, math.clamp(dt*Fly.smoothness,0,1))
    Fly.parts.lv.VectorVelocity=Fly.velocity

    -- If we're sitting in a car, override the car's velocity too so the
    -- whole vehicle flies with us (otherwise wheel springs + body movers
    -- yank it back down).
    local char = LocalPlayer.Character
    local hum  = char and char:FindFirstChildOfClass("Humanoid")
    local seat = hum and hum.SeatPart
    if seat then
        -- Walk up from the seat to the TOP-MOST Model (the car)
        local node, car = seat.Parent, nil
        while node and node ~= workspace do
            if node:IsA("Model") then car = node end
            node = node.Parent
        end
        if car then
            for _, p in ipairs(car:GetDescendants()) do
                if p:IsA("BasePart") then
                    pcall(function()
                        p.AssemblyLinearVelocity  = Fly.velocity
                        p.AssemblyAngularVelocity = Vector3.zero
                    end)
                end
            end
        end
    end

    local flat=Vector3.new(cf.LookVector.X,0,cf.LookVector.Z)
    if flat.Magnitude<0.001 then return end
    local yaw=CFrame.lookAt(Vector3.zero,flat.Unit)
    local lv2=yaw:VectorToObjectSpace(Fly.velocity); local mtr=math.rad(Fly.maxTilt)
    Fly.parts.ao.CFrame=yaw*CFrame.Angles(math.clamp(lv2.Z/Fly.speed,-1,1)*mtr,0,-math.clamp(lv2.X/Fly.speed,-1,1)*mtr)
end))

local Noclip={enabled=false,conn=nil}
local function noclipSet(v)
    Noclip.enabled=v
    if v and not Noclip.conn then
        Noclip.conn=RunService.Stepped:Connect(function()
            local char=LocalPlayer.Character; if not char then return end
            for _,p in ipairs(char:GetDescendants()) do
                if p:IsA("BasePart") then p.CanCollide=false end
            end
        end)
    elseif not v then
        if Noclip.conn then Noclip.conn:Disconnect(); Noclip.conn=nil end
        local char=LocalPlayer.Character
        if char then for _,p in ipairs(char:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide=true end
        end end
    end
end

local Fullbright={enabled=false,saved=nil}
local function fullbrightSet(v)
    if v==Fullbright.enabled then return end
    if v then
        Fullbright.saved={Brightness=Lighting.Brightness,FogEnd=Lighting.FogEnd,
            GlobalShadows=Lighting.GlobalShadows,Ambient=Lighting.Ambient,OutdoorAmbient=Lighting.OutdoorAmbient}
        Lighting.Brightness=2; Lighting.FogEnd=1e9; Lighting.GlobalShadows=false
        Lighting.Ambient=Color3.new(1,1,1); Lighting.OutdoorAmbient=Color3.new(1,1,1)
    elseif Fullbright.saved then
        for k,val in pairs(Fullbright.saved) do Lighting[k]=val end; Fullbright.saved=nil
    end
    Fullbright.enabled=v
end

local Keybinds={}
local listeningFor=nil
local function registerKeybind(name,defaultKey,callback) Keybinds[name]={key=defaultKey,callback=callback} end
local function rebindKeybind(name,newKey) if Keybinds[name] then Keybinds[name].key=newKey end end

bind(UserInputService.InputBegan:Connect(function(input,processed)
    if processed and not listeningFor then return end
    if input.UserInputType~=Enum.UserInputType.Keyboard then return end
    if listeningFor then
        local cb=listeningFor.commit; listeningFor=nil
        if input.KeyCode~=Enum.KeyCode.Escape then cb(input.KeyCode) else cb(nil) end; return
    end
    for _,kb in pairs(Keybinds) do if input.KeyCode==kb.key then kb.callback() end end
end))

        tween(btn,0.16,{BackgroundTransparency=1, TextColor3=Theme.TextDim})
    end)

local function makeTitleRow(card,labelText,initial,onChange,order)
    local row=makeRowFrame(card,order or 1,24)
    local label=Instance.new("TextLabel")
    label.Size=UDim2.new(1,-60,1,0); label.BackgroundTransparency=1
    label.Text=labelText; label.Font=Fonts.Bold; label.TextSize=13
    label.TextXAlignment=Enum.TextXAlignment.Left; label.Parent=row
    themeify(label,"TextColor3","Text")
    if initial==nil then return {row=row} end

    -- Track
    local box=Instance.new("Frame")
    box.AnchorPoint=Vector2.new(1,0.5); box.Position=UDim2.new(1,0,0.5,0)
    box.Size=UDim2.fromOffset(34,18); box.BorderSizePixel=0; box.Parent=row
    themeify(box,"BackgroundColor3",initial and "Accent" or "SurfaceAlt"); corner(box,3)

    -- Knob (flat square with subtle rounding)
    local knob=Instance.new("Frame")
    knob.Size=UDim2.fromOffset(12,12)
    knob.Position=initial and UDim2.new(1,-15,0.5,-6) or UDim2.new(0,3,0.5,-6)
    knob.BorderSizePixel=0; knob.Parent=box
    knob.BackgroundColor3 = Color3.fromRGB(245, 238, 225)  -- pale cream
    corner(knob,2)

    local btn=Instance.new("TextButton")
    btn.Size=UDim2.fromScale(1,1); btn.BackgroundTransparency=1; btn.Text=""; btn.Parent=box

    local state=initial
    local function setState(s)
        state=s
        tween(box,0.14,{BackgroundColor3=s and Theme.Accent or Theme.SurfaceAlt})
        tween(knob,0.14,{Position=s and UDim2.new(1,-15,0.5,-6) or UDim2.new(0,3,0.5,-6)})
        retagTheme(box,"BackgroundColor3",s and "Accent" or "SurfaceAlt")
        onChange(s)
    end
    btn.MouseButton1Click:Connect(function() setState(not state) end)
    return {row=row,set=setState,get=function() return state end}
end

local function makeSliderRow(card,default,minV,maxV,onChange,order)
    local row=makeRowFrame(card,order or 2,24)
    local trackBar=Instance.new("Frame")
    trackBar.AnchorPoint=Vector2.new(0,0.5); trackBar.Position=UDim2.new(0,0,0.5,0)
    trackBar.Size=UDim2.new(1,-62,0,6); trackBar.BorderSizePixel=0; trackBar.Parent=row
    themeify(trackBar,"BackgroundColor3","SurfaceAlt"); corner(trackBar,3)

    local fill=Instance.new("Frame")
    fill.Size=UDim2.new((default-minV)/(maxV-minV),0,1,0); fill.BorderSizePixel=0; fill.Parent=trackBar
    themeify(fill,"BackgroundColor3","Accent"); corner(fill,3)

    -- Knob — simple solid square with subtle rounding
    local knob=Instance.new("Frame")
    knob.AnchorPoint=Vector2.new(0.5,0.5); knob.Position=UDim2.new(1,0,0.5,0)
    knob.Size=UDim2.fromOffset(12,12); knob.BorderSizePixel=0; knob.Parent=fill
    themeify(knob,"BackgroundColor3","Accent"); corner(knob,3)

    local valueLabel=Instance.new("TextLabel")
    valueLabel.AnchorPoint=Vector2.new(1,0.5); valueLabel.Position=UDim2.new(1,0,0.5,0)
    valueLabel.Size=UDim2.fromOffset(54,22); valueLabel.Text=tostring(default)
    valueLabel.Font=Fonts.Mono; valueLabel.TextSize=12; valueLabel.Parent=row
    themeify(valueLabel,"BackgroundColor3","SurfaceAlt"); themeify(valueLabel,"TextColor3","Text"); corner(valueLabel,5)
    local valueStroke=stroke(valueLabel,Theme.Border,1); valueStroke.Transparency=0.5
    table.insert(themed,{valueStroke,"Color","Border"})

    local dragging=false
    local function update(x)
        local pct=math.clamp((x-trackBar.AbsolutePosition.X)/trackBar.AbsoluteSize.X,0,1)
        -- Smooth tween on the fill width — feels fluid even on sharp drags
        tween(fill,0.08,{Size = UDim2.new(pct,0,1,0)})
        local v=math.floor(minV+(maxV-minV)*pct+0.5)
        valueLabel.Text=tostring(v); onChange(v)
    end
    local function setVisual(v)
        tween(fill,0.18,{Size = UDim2.new((v-minV)/(maxV-minV),0,1,0)})
        valueLabel.Text=tostring(v)
    end
    trackBar.InputBegan:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then
            dragging=true; update(i.Position.X) end
    end)
    knob.InputBegan:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then
            dragging=true end
    end)
    bind(UserInputService.InputChanged:Connect(function(i)
        if dragging and (i.UserInputType==Enum.UserInputType.MouseMovement or i.UserInputType==Enum.UserInputType.Touch) then
            update(i.Position.X) end
    end))
    bind(UserInputService.InputEnded:Connect(function(i)
        if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then
            dragging=false end
    end))
    return {setVisual=setVisual}
end

        o.Text="  "..opt; o.TextXAlignment=Enum.TextXAlignment.Left

        o.MouseButton1Click:Connect(function()
            btn.Text=" "..opt.."         ▾"; list.Visible=false; onChange(opt)

        if list.Visible then list.Visible=false; return end

            return mx>=p.X and mx<=p.X+s.X and my>=p.Y and my<=p.Y+s.Y end
        if not within(list) and not within(btn) then list.Visible=false end
    end))
    return btn
end

local Pages={}; local activeTab=nil

local function switchTab(name)
    activeTab=name
    for n,p in pairs(Pages) do
        p.frame.Visible=(n==name)
        if n==name then
            p.button.BackgroundTransparency=0
            tween(p.button,0.12,{BackgroundColor3=Theme.SurfaceAlt,TextColor3=Theme.Text})
            retagTheme(p.button,"BackgroundColor3","SurfaceAlt"); retagTheme(p.button,"TextColor3","Text")
            p.indicator.Visible=true
        else
            tween(p.button,0.14,{BackgroundTransparency=1, TextColor3=Theme.TextDim})
            retagTheme(p.button,"BackgroundColor3","Background"); retagTheme(p.button,"TextColor3","TextDim")
            p.indicator.Visible=false
        end
    end
end

local function addTab(name,order)
    local page=makePage()
    local b=Instance.new("TextButton")
    b.Size=UDim2.new(1,0,0,28); b.AutoButtonColor=false
    b.Text="  "..name; b.TextXAlignment=Enum.TextXAlignment.Left
    b.Font=Fonts.Bold; b.TextSize=12; b.LayoutOrder=order; b.Parent=Sidebar
    b.BackgroundTransparency=1
    themeify(b,"BackgroundColor3","Background"); themeify(b,"TextColor3","TextDim"); corner(b,3)

    -- Simple flat indicator bar on the left edge of the active tab
    local indicator=Instance.new("Frame")
    indicator.Size=UDim2.new(0,2,0.6,0); indicator.Position=UDim2.new(0,2,0.2,0)
    indicator.BorderSizePixel=0; indicator.Visible=false; indicator.Parent=b
    themeify(indicator,"BackgroundColor3","Accent")

    b.MouseEnter:Connect(function()
        if activeTab~=name then
            tween(b,0.1,{BackgroundTransparency=0, BackgroundColor3=Theme.Surface, TextColor3=Theme.Text})
        end
    end)
    b.MouseLeave:Connect(function()
        if activeTab~=name then
            tween(b,0.12,{BackgroundTransparency=1, TextColor3=Theme.TextDim})
        end
    end)
    b.MouseButton1Click:Connect(function()
        b.BackgroundTransparency=0
        switchTab(name)
    end)
    Pages[name]={frame=page,button=b,indicator=indicator}
    return page
end

        table.insert(statusLog, 1, msg)
        while #statusLog > 5 do table.remove(statusLog) end
        if statusLabel then statusLabel.Text = table.concat(statusLog, "\n") end
        print("[Drag]", msg)
    end

    -- ── Find drag remote (with debug) ───────────────────────────
    local dragRemote
    do
        local interaction = ReplicatedStorage:FindFirstChild("Interaction")
        if interaction then
            dragRemote = interaction:FindFirstChild("ClientIsDragging")
        end
        if not dragRemote then
            task.spawn(function()
                interaction = ReplicatedStorage:WaitForChild("Interaction", 10)
                if interaction then
                    dragRemote = interaction:WaitForChild("ClientIsDragging", 10)
                end
                if dragRemote then
                    log("✓ dragRemote found")
                else
                    log("✗ dragRemote NOT found")
                end
            end)
        end
    end

    -- ── Selection state ─────────────────────────────────────────
    local selected = {}   -- [item] = highlight
    local selectionEnabled = false
    local selCountLabel

    local function updateCount()
        if selCountLabel then
            local n = 0; for _ in pairs(selected) do n += 1 end
            selCountLabel.Text = "Selected: " .. n
        end
    end

    -- Walk up to find the LT2 Model
    local function resolveItem(part)
        if not part then return end
        local cur = part
        while cur and cur ~= workspace do
            if cur:IsA("Model") then
                if cur:FindFirstChild("Owner")
                or cur:FindFirstChild("WoodSection")
                or cur:FindFirstChild("TreeClass")
                or cur:FindFirstChild("BoxItemName")
                or cur:FindFirstChild("PurchasedBoxItemName")
                or cur:FindFirstChild("DraggableItem")
                or cur:FindFirstChild("Handle") then
                    return cur
                end
            end
            cur = cur.Parent
        end
        return part.Parent
    end

    local function highlight(item)
        if not item or selected[item] then return end
        local h = Instance.new("Highlight")
        h.FillColor       = Color3.fromRGB(70, 200, 130)
        h.OutlineColor    = Color3.fromRGB(120, 255, 180)
        h.FillTransparency = 0.6
        h.Adornee = item
        h.Parent = item
        selected[item] = h
        updateCount()

        -- Debug: log ownership info
        local owner = item:FindFirstChild("Owner")
        local ownerName = "(no owner)"
        if owner and owner.Value then ownerName = tostring(owner.Value) end
        log("Selected: " .. item.Name .. " | Owner: " .. ownerName)
    end

    local function unhighlight(item)
        if not item then return end
        local h = selected[item]
        if h then pcall(function() h:Destroy() end) end
        selected[item] = nil
        updateCount()
    end

    local function clearAll()
        for _, h in pairs(selected) do
            if h then pcall(function() h:Destroy() end) end
        end
        table.clear(selected); updateCount()
        log("Selection cleared")
    end

    -- ── THE TELEPORT (Kykyry3ina's bringy pattern) ────────────────
    -- For each item:
    --   1. Save player CFrame (origin)
    --   2. Teleport player ABOVE the item's main part
    --   3. Spawn a parallel coroutine that spams ClientIsDragging
    --      every frame for 60 frames (claims & holds ownership)
    --   4. In the main thread, set the item's PrimaryPart, then
    --      PivotTo(originCF) 60 times zeroing velocity each frame
    --   5. The two run in parallel — the spam keeps server-side
    --      ownership alive while PivotTo replicates the position
    -- This is the SAME pattern script 2 uses for bringing a cut tree.

    local function getMainPart(item)
        if not item or not item.Parent then return end
        if item:IsA("BasePart") then return item end
        if item:IsA("Model") then
            -- Prefer WoodSection (logs / trees), then PrimaryPart, then anything
            local ws = item:FindFirstChild("WoodSection")
            if ws and ws:IsA("BasePart") then return ws end
            if item.PrimaryPart then return item.PrimaryPart end
            return item:FindFirstChildWhichIsA("BasePart", true)
        end
    end

    -- TP iterations: 20 frames (~0.33s) — fast but gives the server a
    -- moment to settle the drag claim before the PivotTo loop starts.
    local TP_ITERS = 20

    local function tpOne(item, originCF)
        if not item or not item.Parent then return false, "Item gone" end
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp or not dragRemote then return false, "No HRP/remote" end

        local mainPart = getMainPart(item)
        if not mainPart then return false, "No part" end

        -- Step 1: Teleport ABOVE the item. Bumped from 0.03→0.1s — too
        -- short and the player's new position hasn't replicated yet, so
        -- the server rejects our drag claim and the PivotTo writes don't
        -- stick (that's why items got "left behind" on fast TPs).
        hrp.CFrame = CFrame.new(mainPart.Position + Vector3.new(0, 5, 0))
        task.wait(0.1)

        -- Step 2: Set primary part for PivotTo
        if item:IsA("Model") and not item.PrimaryPart then
            pcall(function() item.PrimaryPart = mainPart end)
        end

        -- Step 3: Prime the drag claim with a few synchronous fires BEFORE
        -- the parallel spam — gives the server a definite "I'm dragging
        -- this" signal that survives even if the spam task gets delayed.
        for _ = 1, 3 do
            pcall(function() dragRemote:FireServer(item) end)
            task.wait()
        end

        -- Step 4: Parallel spam ClientIsDragging — keeps the claim alive
        task.spawn(function()
            for _ = 1, TP_ITERS do
                if not item or not item.Parent then break end
                pcall(function() dragRemote:FireServer(item) end)
                task.wait()
            end
        end)

        -- Step 5: PivotTo origin — zero BOTH AssemblyLinearVelocity (the
        -- real network-replicated one) AND legacy .Velocity each frame.
        -- Stuff like wood logs has additional rigid body parts; relying on
        -- .Velocity alone leaves residual motion that the server snaps back.
        local function zeroPart(p)
            pcall(function() p.AssemblyLinearVelocity  = Vector3.zero end)
            pcall(function() p.AssemblyAngularVelocity = Vector3.zero end)
            pcall(function() p.Velocity                = Vector3.zero end)
        end
        for _ = 1, TP_ITERS do
            if not item or not item.Parent then break end
            if item:IsA("Model") and item.PrimaryPart then
                zeroPart(item.PrimaryPart)
                pcall(function() item:PivotTo(originCF) end)
            elseif item:IsA("BasePart") then
                zeroPart(item)
                pcall(function() item.CFrame = originCF end)
            end
            task.wait()
        end

        -- Step 6: Settle pass — a few extra writes after the spam ends so
        -- the final position is the one the server commits, not whatever
        -- momentum was active when the loop exited.
        for _ = 1, 5 do
            if not item or not item.Parent then break end
            if item:IsA("Model") and item.PrimaryPart then
                zeroPart(item.PrimaryPart)
                pcall(function() item:PivotTo(originCF) end)
            elseif item:IsA("BasePart") then
                zeroPart(item)
                pcall(function() item.CFrame = originCF end)
            end
            task.wait()
        end
        return true
    end

    local function tpSelected()
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then log("✗ No HRP"); return end

        local items = {}
        for item in pairs(selected) do
            if item and item.Parent then table.insert(items, item) end
        end
        if #items == 0 then log("✗ Nothing selected"); return end

        local originCF = hrp.CFrame
        log("Shuttling " .. #items .. " items…")

        task.spawn(function()
            local ok = 0
            for i, item in ipairs(items) do
                local success, err = tpOne(item, originCF)
                if success then
                    ok += 1
                    log("✓ Brought " .. item.Name)
                else
                    log("✗ " .. item.Name .. ": " .. tostring(err))
                end
                -- Return to origin between items so we stay near home
                local c = LocalPlayer.Character
                local h = c and c:FindFirstChild("HumanoidRootPart")
                if h then h.CFrame = originCF end
                task.wait()
            end
            log("✓ Done — " .. ok .. "/" .. #items .. " brought home")
        end)
    end

    -- ── Auto Sell: stream every Plank/Log on the plot to WoodRUs ────
    -- Uses the exact pattern from the verified LT2 admin GUI on Pastebin:
    --   1) Spawn one coroutine per WoodSection — 10 iters, set CFrame to
    --      (315, -0.296, 85.791) * Angles(rad(90),0,0) (the conveyor lay-flat).
    --   2) Spawn a second coroutine per Plank/Log — 20 iters of
    --      ClientIsDragging:FireServer to keep the server claim alive.
    -- The 90° X rotation is what stops planks glitching through the floor
    -- or shooting above the roof — it lays them parallel to the conveyor.
    local SELL_POS = Vector3.new(315, -0.296, 85.791)
    local SELL_CF  = CFrame.new(SELL_POS) * CFrame.Angles(math.rad(90), 0, 0)

    -- Auto-sell often fails because the section bounces off the conveyor
    -- or Roblox physics freezes the conveyor (known LT2 bug). Stronger
    -- pattern: longer iteration counts (50/60), zero AssemblyLinear &
    -- AssemblyAngular velocity each frame so the section doesn't drift,
    -- and slightly stagger fires so the server processes each in order.
    local function tpSellSection(section, owner)
        task.spawn(function()
            for _ = 1, 50 do
                if not section or not section.Parent then break end
                pcall(function()
                    section.AssemblyLinearVelocity  = Vector3.zero
                    section.AssemblyAngularVelocity = Vector3.zero
                    section.CFrame = SELL_CF
                end)
                task.wait()
            end
        end)
        task.spawn(function()
            for _ = 1, 60 do
                if not owner or not owner.Parent then break end
                if dragRemote then
                    pcall(function() dragRemote:FireServer(owner) end)
                end
                task.wait()
            end
        end)
    end

    -- Destroy every weld/constraint inside a Model so each section
    -- becomes physically independent (so they don't all bounce off the
    -- conveyor together as one welded chunk).
    local function nukeWelds(model)
        local n = 0
        for _, d in ipairs(model:GetDescendants()) do
            if d:IsA("WeldConstraint") or d:IsA("Weld") or d:IsA("Snap")
            or d:IsA("Glue") or d:IsA("ManualWeld") or d:IsA("Motor")
            or d:IsA("Motor6D") or d:IsA("JointInstance") then
                pcall(function() d:Destroy() end)
                n = n + 1
            end
        end
        return n
    end

    local function autoSellAll()
        if not dragRemote then log("✗ dragRemote missing"); return end
        local playerModels = workspace:FindFirstChild("PlayerModels")
        local logModels    = workspace:FindFirstChild("LogModels")
        if not playerModels and not logModels then
            log("✗ PlayerModels + LogModels missing"); return
        end
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then log("✗ No HRP"); return end

        -- Collect owned LOG MODELS (whole-log targets — exactly how
        -- TP All Wood works, just shipping them to SELL_CF instead of
        -- back to player). Nukes welds first so they sell cleanly.
        local logs = {}
        local plankCount, logCount, totalWelds = 0, 0, 0

        if playerModels then
            for _, m in ipairs(playerModels:GetChildren()) do
                local own = m:FindFirstChild("Owner")
                if own and own.Value == LocalPlayer then
                    if m.Name == "Plank" then
                        plankCount = plankCount + 1
                        totalWelds = totalWelds + nukeWelds(m)
                        table.insert(logs, m)
                    elseif m.Name:match("^Loose_") or m:FindFirstChild("WoodSection") then
                        logCount = logCount + 1
                        totalWelds = totalWelds + nukeWelds(m)
                        table.insert(logs, m)
                    end
                end
            end
        end
        if logModels then
            for _, m in ipairs(logModels:GetChildren()) do
                local own = m:FindFirstChild("Owner")
                if own and own.Value == LocalPlayer then
                    logCount = logCount + 1
                    totalWelds = totalWelds + nukeWelds(m)
                    table.insert(logs, m)
                end
            end
        end

        if #logs == 0 then log("✗ Nothing of yours found"); return end

        local originCF = hrp.CFrame
        log("Auto-selling " .. #logs .. " items (welds nuked: "
            .. totalWelds .. ") — tpOne each to conveyor")

        -- Same loop as TP All Wood: tpOne each LOG MODEL sequentially,
        -- TP player back to origin between each so we don't drift.
        task.spawn(function()
            local sold = 0
            for i, m in ipairs(logs) do
                if m and m.Parent then
                    if tpOne(m, SELL_CF) then sold = sold + 1 end
                end
                local c = LocalPlayer.Character
                local h = c and c:FindFirstChild("HumanoidRootPart")
                if h then h.CFrame = originCF end
                task.wait()
            end
            log("✓ Auto-sell done — " .. sold .. "/" .. #logs
                .. " (" .. plankCount .. " planks + " .. logCount .. " logs)")
        end)
    end

    -- Selection-only sell — sends just the highlighted items.
    local function tpSellSelected()
        if not dragRemote then log("✗ dragRemote missing"); return end
        local items = {}
        for item in pairs(selected) do
            if item and item.Parent then table.insert(items, item) end
        end
        if #items == 0 then log("✗ Nothing selected"); return end

        log("Selling " .. #items .. " selected items…")
        for _, m in ipairs(items) do
            local sections = {}
            if m:IsA("Model") then
                for _, d in ipairs(m:GetDescendants()) do
                    if d.Name == "WoodSection" then table.insert(sections, d) end
                end
            elseif m:IsA("BasePart") then
                table.insert(sections, m)
            end
            for _, sec in ipairs(sections) do tpSellSection(sec, m) end
        end
        log("✓ Streaming " .. #items .. " items to sell zone")
    end

    -- ── Hard Drag: continuous ClientIsDragging spam ─────────────
    -- The drag script's internal loop fires ClientIsDragging every 0.5s.
    -- When it breaks on heavy items, those fires stop and the server
    -- times out the drag in ~2s.  We continuously fire ClientIsDragging
    -- ourselves while the Dragger Part exists in workspace and the mouse
    -- is held, so the server-side drag stays alive forever.
    --
    -- This doesn't require hookfunction or upvalue access — it just
    -- keeps refreshing the server's drag ownership.
    local hardDragEnabled = true
    local mouseHeld = false

    bind(UserInputService.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1
        or i.UserInputType == Enum.UserInputType.Touch then mouseHeld = true end
    end))
    bind(UserInputService.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1
        or i.UserInputType == Enum.UserInputType.Touch then mouseHeld = false end
    end))

    -- Identify the LT2 Dragger Part by its unique signature
    -- (BodyPosition + BodyGyro + Weld children all together).
    local function isDragger(obj)
        return obj and obj:IsA("BasePart")
            and obj:FindFirstChildOfClass("BodyPosition")
            and obj:FindFirstChildOfClass("BodyGyro")
            and obj:FindFirstChildOfClass("Weld")
    end
    local function findActiveDragger()
        for _, obj in ipairs(workspace:GetChildren()) do
            if isDragger(obj) then return obj end
        end
    end

    -- Find the Model the dragger is welded to (walk up from weld.Part1)
    local function getDraggedModel(dragger)
        local weld = dragger:FindFirstChildOfClass("Weld")
        if not weld or not weld.Part1 then return nil end
        local cur = weld.Part1
        while cur and cur ~= workspace do
            if cur:IsA("Model") and (cur:FindFirstChild("Owner")
            or cur:FindFirstChild("WoodSection")
            or cur:FindFirstChild("TreeClass")) then
                return cur
            end
            cur = cur.Parent
        end
        return weld.Part1
    end

    -- Massless mode: when on, every part of the dragged item is set to
    -- Massless = true so the drag tool can move arbitrarily heavy items.
    -- We track which parts we modified per-item so we can restore them.
    local masslessEnabled  = false
    local masslessTracked  = {}  -- [model] = { {part=..., wasMassless=...}, ... }

    local function applyMassless(model)
        if not model or masslessTracked[model] then return end
        local list = {}
        for _, p in ipairs(model:GetDescendants()) do
            if p:IsA("BasePart") then
                table.insert(list, {part = p, wasMassless = p.Massless})
                pcall(function() p.Massless = true end)
            end
        end
        masslessTracked[model] = list
    end
    local function restoreMassless(model)
        local list = masslessTracked[model]
        if not list then return end
        for _, e in ipairs(list) do
            if e.part and e.part.Parent then
                pcall(function() e.part.Massless = e.wasMassless end)
            end
        end
        masslessTracked[model] = nil
    end

    -- Buff the dragger's body movers so the drag tool's internal force
    -- limit doesn't choke on heavy items. We crank MaxForce/MaxTorque
    -- to math.huge and stiffen P/D so positioning is snappier.
    local buffedDraggers = setmetatable({}, {__mode = "k"})
    local function buffDragger(dragger)
        if not dragger or buffedDraggers[dragger] then return end
        buffedDraggers[dragger] = true
        local bp = dragger:FindFirstChildOfClass("BodyPosition")
        local bg = dragger:FindFirstChildOfClass("BodyGyro")
        if bp then
            bp.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            bp.P = 1e5; bp.D = 1250
        end
        if bg then
            bg.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
            bg.P = 1e5; bg.D = 250
        end
        if masslessEnabled then
            local m = getDraggedModel(dragger)
            if m then applyMassless(m) end
        end
    end

    -- Track which item each dragger is moving so we can force-replicate
    -- its final position to the server when the drag ENDS. Without this,
    -- the item visually moves on your screen but snaps back the moment
    -- the server takes over (which is what "if u jump on the item it
    -- just goes back to where u put it" was — pure client-side movement).
    local draggerToItem = setmetatable({}, {__mode = "k"})

    -- Force the server to accept the item's current client-side position.
    -- Same pattern script 2 uses to bring a cut tree home: spam
    -- ClientIsDragging in parallel while PivotTo'ing every frame.
    local function commitDragPosition(item)
        if not item or not item.Parent then return end
        if not dragRemote then return end

        local mainPart
        if item:IsA("BasePart") then mainPart = item
        elseif item:IsA("Model") then
            mainPart = item:FindFirstChild("WoodSection")
                    or item.PrimaryPart
                    or item:FindFirstChildWhichIsA("BasePart", true)
            if mainPart and not item.PrimaryPart then
                pcall(function() item.PrimaryPart = mainPart end)
            end
        end
        if not mainPart then return end

        -- Snapshot the CFrame the client THINKS the item is at right now
        local lockCF = item:IsA("Model") and item:GetPivot() or mainPart.CFrame

        task.spawn(function()
            -- Parallel ownership spam (30 frames)
            task.spawn(function()
                for _ = 1, 30 do
                    if not item or not item.Parent then break end
                    pcall(function() dragRemote:FireServer(item) end)
                    task.wait()
                end
            end)
            -- PivotTo loop forces server to accept this position (30 frames)
            for _ = 1, 30 do
                if not item or not item.Parent then break end
                if item:IsA("Model") and item.PrimaryPart then
                    pcall(function() item.PrimaryPart.Velocity = Vector3.zero end)
                    pcall(function() item:PivotTo(lockCF) end)
                elseif item:IsA("BasePart") then
                    pcall(function() item.Velocity = Vector3.zero end)
                    pcall(function() item.CFrame   = lockCF end)
                end
                task.wait()
            end
        end)
    end

    bind(workspace.ChildAdded:Connect(function(child)
        if not hardDragEnabled then return end
        task.wait()
        if isDragger(child) then
            buffDragger(child)
            -- Remember which item this dragger is moving — we need it
            -- after the dragger is destroyed (drag ends).
            local item = getDraggedModel(child)
            if item then draggerToItem[child] = item end

            child.AncestryChanged:Connect(function(_, parent)
                if parent == nil then
                    local trackedItem = draggerToItem[child]
                    -- 1) Restore mass first so PivotTo physics make sense
                    for model in pairs(masslessTracked) do restoreMassless(model) end
                    buffedDraggers[child] = nil
                    -- 2) Commit final position to server (fixes the snap-back)
                    if trackedItem and hardDragEnabled then
                        commitDragPosition(trackedItem)
                    end
                    draggerToItem[child] = nil
                end
            end)
        end
    end))

    -- Heartbeat: fires drag remote every 0.08s while mouse held — keeps the
    -- server ownership claim alive AND re-buffs the dragger every frame.
    local dragTimer = 0
    bind(RunService.Heartbeat:Connect(function(dt)
        if not hardDragEnabled or not mouseHeld or not dragRemote then dragTimer = 0; return end
        dragTimer = dragTimer + dt

        local dragger = findActiveDragger()
        if dragger then
            buffDragger(dragger)              -- re-buff in case the game reset values
        end

        if dragTimer < 0.08 then return end
        dragTimer = 0
        if not dragger then return end

        local item = getDraggedModel(dragger)
        if not item then return end
        pcall(function() dragRemote:FireServer(item) end)

        -- Also fire ClientRequestOwnership for a stronger claim if available
        local rs = game:GetService("ReplicatedStorage")
        local interaction = rs:FindFirstChild("Interaction")
        local reqOwn = interaction and interaction:FindFirstChild("ClientRequestOwnership")
        if reqOwn then
            local weld = dragger:FindFirstChildOfClass("Weld")
            if weld and weld.Part1 then
                pcall(function() reqOwn:FireServer(weld.Part1) end)
            end
        end
    end))

    -- ── Group Select + click handler ────────────────────────────
    -- Wrapped in a `do ... end` so all the helper locals release
    -- after setup (Luau has a 200-register-per-function limit and
    -- the page builder is already crowded). The setters/getters
    -- below are returned through `groupAPI` so the click handler
    -- and the toggle/slider UI can read/write the state.
    local groupAPI
    do
        local enabled = false
        local radius  = 30

        local function itemAnchor(item)
            if item:IsA("BasePart") then return item end
            if item:IsA("Model") then
                return item.PrimaryPart
                    or item:FindFirstChild("WoodSection")
                    or item:FindFirstChild("Main")
                    or item:FindFirstChild("Handle")
                    or item:FindFirstChildWhichIsA("BasePart", true)
            end
        end

        local function selectGroup(originItem)
            local anchor = itemAnchor(originItem)
            if not anchor then return 0 end
            local key = originItem.Name
            local centerPos = anchor.Position
            local r2 = radius * radius
            local n = 0
            local roots = {}
            for _, fname in ipairs({"LogModels", "PlayerModels", "Stores"}) do
                local f = workspace:FindFirstChild(fname)
                if f then table.insert(roots, f) end
            end
            if #roots == 0 then roots = {workspace} end
            for _, root in ipairs(roots) do
                for _, m in ipairs(root:GetChildren()) do
                    if m:IsA("Model") and m.Name == key and m ~= originItem
                    and not selected[m] then
                        local a = itemAnchor(m)
                        if a then
                            local d = a.Position - centerPos
                            if d:Dot(d) <= r2 then
                                highlight(m)
                                n = n + 1
                            end
                        end
                    end
                end
            end
            return n
        end

        groupAPI = {
            setEnabled = function(v) enabled = v end,
            setRadius  = function(v) radius = v end,
            getRadius  = function() return radius end,
            isEnabled  = function() return enabled end,
            selectGroup = selectGroup,
        }

        -- Click handler defined HERE inside the do block so it captures
        -- the local closure (enabled/radius/selectGroup) directly without
        -- consuming an outer-scope register.
        bind(UserInputService.InputBegan:Connect(function(input, processed)
            if processed or not selectionEnabled then return end
            local mouse = LocalPlayer:GetMouse()
            if not mouse or not mouse.Target then return end
            local item = resolveItem(mouse.Target)
            if not item then return end
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                highlight(item)
                if enabled then
                    local n = selectGroup(item)
                    if n > 0 then
                        log("Group select: +" .. n .. " '" .. item.Name
                            .. "' within " .. radius .. " studs")

    -- Returns the best axe in backpack+character for the given tree class.
    -- Tries TWO known LT2 module paths (script 2 uses ReplicatedStorage.AxeClasses,

    -- path resolves stats, the tool is STILL accepted — having any tool with
    -- a ToolName that isn't BlueprintTool counts as a valid axe.
    local function getBestAxeLT2(forClass)
        forClass = forClass or "Generic"
        local rs         = game:GetService("ReplicatedStorage")
        local axeClasses = rs:FindFirstChild("AxeClasses")
        local axeFolder  = rs:FindFirstChild("Purchasables")
        if axeFolder then axeFolder = axeFolder:FindFirstChild("Tools")    end
        if axeFolder then axeFolder = axeFolder:FindFirstChild("AllTools") end

        local best, bestDmg = nil, -1
        local function rank(t)
            if t.Name == "BlueprintTool" then return end
            local tn = t:FindFirstChild("ToolName")
            if not tn then return end
            -- Try to read damage from either module path; fall back to 0
            local stats
            if axeClasses then
                local mod = axeClasses:FindFirstChild("AxeClass_" .. tn.Value)
                if mod then
                    local ok, s = pcall(function() return require(mod).new() end)
                    if ok then stats = s end
                end
            end
            if not stats and axeFolder then
                local tf = axeFolder:FindFirstChild(tn.Value)
                if tf and tf:FindFirstChild("AxeClass") then
                    local ok, s = pcall(function() return require(tf.AxeClass).new() end)
                    if ok then stats = s end
                end
            end
            local dmg = 0
            if stats then
                if stats.SpecialTrees and stats.SpecialTrees[forClass] then
                    dmg = stats.SpecialTrees[forClass].Damage or stats.Damage or 0
                else
                    dmg = stats.Damage or 0
                end
            end
            -- bestDmg starts at -1 so even unknown-damage tools get picked
            if dmg > bestDmg then bestDmg = dmg; best = t end
        end
        for _, t in ipairs(LocalPlayer.Backpack:GetChildren()) do rank(t) end
        local char = LocalPlayer.Character
        if char then for _, t in ipairs(char:GetChildren()) do
            if t:IsA("Tool") then rank(t) end
        end end
        return best
    end

    local function findBestTree(treeClass)

        for _, region in ipairs(workspace:GetChildren()) do
            if region.Name == "TreeRegion" then
                for _, tree in ipairs(region:GetChildren()) do
                    local tc    = tree:FindFirstChild("TreeClass")
                    local owner = tree:FindFirstChild("Owner")
                    local ws    = tree:FindFirstChild("WoodSection")
                    if tc and owner and ws and tc.Value == treeClass
                    and (owner.Value == nil or owner.Value == LocalPlayer) then
                        local secs = 0
                        for _, v in ipairs(tree:GetDescendants()) do
                            if v.Name=="WoodSection" then secs+=1 end
                        end
                        if secs > bestSec then bestSec=secs; best=tree end
                    end
                end
            end
        end
        return best
    end

    -- Bring Tree — hybrid of script2's bringy() + script4's bringTree.
    -- script2 contributions:
    --   • TP next to WoodSection (+Vec3(5,0,0)) — NOT inside it
    --   • Heartbeat connection that re-pins HRP every frame
    --     (replaces BodyPosition + BodyGyro — simpler, no Humanoid state issues)
    --   • UnequipTools first (avoids "rapid swing" anti-cheat)
    --   • LoneCave hahafunnygod RootJoint clone trick
    -- script4 contributions:
    --   • cooldown = -14 (anti-throttle bypass — chops every frame)
    --   • LogModels.ChildAdded watcher SHUTTLES IN-PLACE on log spawn:
    --     10× drag + Heartbeat:wait + SetPrimaryPartCFrame(oldPos) ONE shot
    --   • repeat wait() chop() until treeIsCut (frame-rate chopping)
    -- This is significantly faster + more reliable than the old version.
    local function cutAndBringTree()
        if bringTreeActive then log("Already running — Stop first"); return end
        local char = LocalPlayer.Character
        if not char then log("No character"); return end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if not hrp then log("No HRP"); return end
        if not dragRemote then log("dragRemote not found"); return end
        local treeClass = selectedTreeClass
        if not treeClass or treeClass == "" then log("Pick a tree class first!"); return end
        local axe = getBestAxeLT2(treeClass)
        if not axe then log("No axe in your backpack"); return end
        local tree = findBestTree(treeClass)
        if not tree then log("No available "..treeClass.." tree found"); return end
        local ws = tree:FindFirstChild("WoodSection")
        if not ws then log("Tree has no WoodSection"); return end

        if treeClass == "LoneCave" and axe.ToolName.Value ~= "EndTimesAxe" then
            log("LoneCave needs the End Times axe"); return
        end

        bringTreeActive = true
        log("Cutting " .. treeClass .. "…")
        local originCF = hrp.CFrame
        local remoteProxy = game:GetService("ReplicatedStorage").Interaction.RemoteProxy
        local cutEvent = tree:FindFirstChild("CutEvent")
        local treeIsCut = false
        local newLog = nil

        task.spawn(function()
            -- 1) Unequip tools first
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then pcall(function() hum:UnequipTools() end) end
            task.wait()

            -- 2) LoneCave RootJoint trick (script2's hahafunnygod)

                pcall(function()
                    local rj = hrp:FindFirstChild("RootJoint")
                    if rj then rj:Clone().Parent = rj.Parent; rj:Destroy() end
                end)
                task.wait()
            end

            -- 3) Freeze the player INSIDE the bottom WoodSection (proven
            --    original lock — user prefers this over the "stand next
            --    to tree" approach from script2).
            --    a) Find the LOWEST WoodSection by world bottom-Y (handles
            --       multi-section trees with twisted trunks)
            --    b) HRP.CanCollide=false so the trunk doesn't shove HRP
            --       out the side every physics tick
            --    c) BodyPosition + BodyGyro lock HRP to that section's
            --       CFrame with infinite force — Humanoid stays in normal
            --       state (not Physics), so CutEvent's server validator
            --       still accepts swings (anchoring HRP would force
            --       Physics state and silently fail).
            --    d) Heartbeat backup re-snaps CFrame each frame using a
            --       LIVE character lookup (survives respawn)
            --    Skipped for LoneCave (RootJoint clone is enough).
            local stayConn, lockBP, lockBG
            local origCanCollide = hrp.CanCollide

            local lowestSec, lowestY = nil, math.huge
            for _, d in ipairs(tree:GetDescendants()) do
                if d:IsA("BasePart") and d.Name:match("^WoodSection") then
                    local botY = d.Position.Y - d.Size.Y * 0.5
                    if botY < lowestY then
                        lowestY   = botY
                        lowestSec = d
                    end
                end
            end
            local cutSec = lowestSec or ws
            local lockCF = cutSec.CFrame

            if treeClass ~= "LoneCave" then
                pcall(function() hrp.CanCollide = false end)

                lockBP = Instance.new("BodyPosition")
                lockBP.Name = "AntiFallLock"
                lockBP.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                lockBP.P = 1e5; lockBP.D = 1250
                lockBP.Position = lockCF.Position
                lockBP.Parent = hrp

                lockBG = Instance.new("BodyGyro")
                lockBG.Name = "AntiFallGyro"
                lockBG.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
                lockBG.P = 1e5; lockBG.D = 250
                lockBG.CFrame = lockCF
                lockBG.Parent = hrp

                stayConn = RunService.Heartbeat:Connect(function()
                    local c = LocalPlayer.Character
                    local h = c and c:FindFirstChild("HumanoidRootPart")
                    if h and h.Parent then
                        h.CFrame = lockCF
                    end
                end)
            end

            -- Initial single TP INTO the cut section; BodyPosition + heartbeat take over
            hrp.CFrame = lockCF
            task.wait(0.15)

            -- 4) LogModels.ChildAdded watcher (script4 pattern): on log
            --    spawn, immediately claim ownership 10× and SetPrimaryPart
            --    to originCF — done in ONE shot inside the watcher so
            --    there's no race against further server updates.
            local logConn
            local logModels = workspace:FindFirstChild("LogModels")
            if logModels then
                logConn = logModels.ChildAdded:Connect(function(child)
                    child:WaitForChild("Owner", 5)
                    if child:FindFirstChild("Owner")
                    and child.Owner.Value == LocalPlayer
                    and child:FindFirstChild("TreeClass")
                    and child.TreeClass.Value == treeClass then
                        newLog = child

                        -- INSTANT ANCHOR every BasePart so the log can't
                        -- fall into the void during the shuttle setup.
                        -- We restore CanCollide/Anchored after the shuttle
                        -- (in phase 9) so the player can interact with it.
                        for _, p in ipairs(child:GetDescendants()) do
                            if p:IsA("BasePart") then
                                pcall(function() p.Anchored = true end)
                            end
                        end

                        -- Drag-spam ownership 10x with Heartbeat waits
                        for _ = 1, 10 do
                            pcall(function() dragRemote:FireServer(child) end)
                            RunService.Heartbeat:Wait()
                        end
                        -- Set PrimaryPart so SetPrimaryPartCFrame can work
                        if not child.PrimaryPart then
                            pcall(function()
                                child.PrimaryPart = child:FindFirstChild("WoodSection")
                                                 or child:FindFirstChildWhichIsA("BasePart", true)
                            end)
                        end
                        -- Shuttle to origin while anchored — bulletproof,
                        -- no gravity, no server snap-back
                        if child.PrimaryPart then
                            pcall(function() child:SetPrimaryPartCFrame(originCF) end)
                        end
                        treeIsCut = true
                    end
                end)
            end

            -- 5) Damage value — fallback HUGE, bark overrides for top trees
            local damage = 999
            pcall(function()
                local rs = game:GetService("ReplicatedStorage")
                local stats
                local axeClasses = rs:FindFirstChild("AxeClasses")
                if axeClasses then
                    local mod = axeClasses:FindFirstChild("AxeClass_"..axe.ToolName.Value)
                    if mod then stats = require(mod).new() end
                end
                if stats then
                    if stats.SpecialTrees and stats.SpecialTrees[treeClass] then
                        damage = stats.SpecialTrees[treeClass].Damage or stats.Damage or damage
                    else
                        damage = stats.Damage or damage
                    end
                end
            end)
            if treeClass == "LoneCave" and axe.ToolName.Value == "EndTimesAxe" then
                damage = 10000000
            elseif treeClass == "Volcano" and axe.ToolName.Value == "FireAxe" then
                damage = 6.35
            end

            local chopArgs = {
                tool         = axe,
                height       = 0.3,
                faceVector   = Vector3.new(1, 0, 0),
                sectionId    = 1,
                hitPoints    = damage,
                cooldown     = -14,
                cuttingClass = "Axe",
            }

            -- 6) Chop every frame (script4 pattern: repeat wait() chop() until cut).
            --    Death-resilient: if the player dies during the chop (LoneCave
            --    crush, Volcano lava, falling damage), wait for respawn and
            --    re-grab the axe + re-establish the in-trunk lock, then RESUME
            --    chopping the SAME tree. Total timeout 90s covers multiple
            --    death cycles for a single LoneCave tree.
            local t0 = tick()
            local trackedChar = LocalPlayer.Character
            repeat
                local liveChar = LocalPlayer.Character

                -- Death detection: Character was destroyed/replaced mid-chop
                if liveChar ~= trackedChar then
                    log("  ⚠ Died mid-chop — recovering…")

                    -- 1) Wait until the new Character exists + HRP is loaded
                    local newChar = LocalPlayer.Character
                    if not newChar or not newChar:FindFirstChild("HumanoidRootPart") then
                        newChar = LocalPlayer.CharacterAdded:Wait()
                    end
                    newChar:WaitForChild("HumanoidRootPart", 10)
                    newChar:WaitForChild("Humanoid", 5)
                    log("  · Respawned; waiting for axe to land in Backpack…")

                    -- 2) RETRY axe-grab for up to 5 seconds. The pickup
                    --    handler in setupLoneCaveAutoPickup is racing us;
                    --    we poll Backpack every 0.2s until an axe shows.
                    local newAxe
                    local axeDeadline = tick() + 5
                    while tick() < axeDeadline do
                        newAxe = getBestAxeLT2(treeClass)
                        if newAxe then break end
                        task.wait(0.2)
                    end
                    if not newAxe then
                        log("  ✗ Axe never returned after 5s — aborting this tree")
                        break
                    end
                    chopArgs.tool = newAxe
                    axe = newAxe
                    log("  · Got axe back: " .. axe.ToolName.Value)

                    -- 3) TP back to the tree base
                    local nHrp = newChar:FindFirstChild("HumanoidRootPart")
                    if nHrp and cutSec and cutSec.Parent then
                        pcall(function() nHrp.CFrame = cutSec.CFrame end)
                        task.wait(0.15)
                        pcall(function() nHrp.CFrame = cutSec.CFrame end)
                        log("  · TP'd back to tree")
                    end

                    -- 4) Re-establish the in-trunk HRP lock for non-LoneCave
                    if treeClass ~= "LoneCave" and cutSec and cutSec.Parent then
                        pcall(function() nHrp.CanCollide = false end)
                        if lockBP and lockBP.Parent then lockBP:Destroy() end
                        if lockBG and lockBG.Parent then lockBG:Destroy() end
                        lockBP = Instance.new("BodyPosition")
                        lockBP.Name = "AntiFallLock"
                        lockBP.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                        lockBP.P = 1e5; lockBP.D = 1250
                        lockBP.Position = cutSec.CFrame.Position
                        lockBP.Parent = nHrp
                        lockBG = Instance.new("BodyGyro")
                        lockBG.Name = "AntiFallGyro"
                        lockBG.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
                        lockBG.P = 1e5; lockBG.D = 250
                        lockBG.CFrame = cutSec.CFrame
                        lockBG.Parent = nHrp
                    end

                    trackedChar = newChar
                    log("  ✓ Resuming chop on same tree — total elapsed "
                        .. math.floor(tick() - t0) .. "s")
                end

                if cutEvent and cutEvent.Parent then
                    pcall(function() remoteProxy:FireServer(cutEvent, chopArgs) end)
                end
                task.wait()
            until treeIsCut or (tick() - t0) > 120 or not bringTreeActive
              or (cutEvent and not cutEvent.Parent)  -- tree got destroyed externally

            -- 7) Cleanup hooks AND release the in-trunk lock — must happen
            --    before the post-cut shuttle so PivotTo can actually move
            --    the player; the BodyPosition would fight the shuttle.
            if stayConn then stayConn:Disconnect(); stayConn = nil end
            if logConn  then logConn:Disconnect();  logConn  = nil end
            if lockBP and lockBP.Parent then lockBP:Destroy() end
            if lockBG and lockBG.Parent then lockBG:Destroy() end
            local cnow = LocalPlayer.Character
            local hnow = cnow and cnow:FindFirstChild("HumanoidRootPart")
            if hnow then pcall(function() hnow.CanCollide = origCanCollide end) end

            if not treeIsCut then
                log("Cut timed out (" .. math.floor(tick() - t0) .. "s)")
                bringTreeActive = false
                return
            end

            -- 8) Settle pass + position verify. Log was already anchored
            --    in the LogModels.ChildAdded watcher so it can't have
            --    fallen. PivotTo it again here in case the in-watcher
            --    shuttle missed, then verify it's actually at origin.
            --    Retry up to 5 times if not.
            task.wait(0.15)
            if newLog and newLog.Parent then
                if not newLog.PrimaryPart then
                    pcall(function()
                        newLog.PrimaryPart = newLog:FindFirstChild("WoodSection")
                                          or newLog:FindFirstChildWhichIsA("BasePart", true)
                    end)
                end

                -- Drag-spam in parallel (server-side ownership claim)
                task.spawn(function()
                    for _ = 1, 60 do
                        if not newLog or not newLog.Parent then break end
                        pcall(function() dragRemote:FireServer(newLog) end)
                        task.wait()
                    end
                end)

                -- Anchor-PivotTo retry loop. Anchored parts can be
                -- PivotTo'd freely without gravity fighting back. Up
                -- to 5 retries with verification.
                local shuttleOk = false
                for attempt = 1, 5 do
                    if not newLog or not newLog.Parent then break end
                    -- Re-anchor (server may have unanchored on its end)
                    for _, p in ipairs(newLog:GetDescendants()) do
                        if p:IsA("BasePart") then
                            pcall(function()
                                p.Anchored = true
                                p.AssemblyLinearVelocity  = Vector3.zero
                                p.AssemblyAngularVelocity = Vector3.zero
                            end)
                        end
                    end
                    -- PivotTo origin
                    if newLog.PrimaryPart then
                        pcall(function() newLog:PivotTo(originCF) end)
                    end
                    task.wait(0.15)
                    -- Verify within 10 studs of origin
                    local pp = newLog.PrimaryPart or newLog:FindFirstChild("WoodSection")
                    if pp and (pp.Position - originCF.Position).Magnitude < 10 then
                        shuttleOk = true
                        log("  ✓ Log shuttled in " .. attempt .. " attempt(s)")
                        break
                    end
                end
                if not shuttleOk then
                    log("  ⚠ Shuttle didn't quite land — log is still anchored,"
                        .. " manually move it or it'll stay where it is")
                end

                -- 9) Unanchor so player can drag/sell normally. Velocity
                --    already zero from above so no surprise drop.
                for _, p in ipairs(newLog:GetDescendants()) do
                    if p:IsA("BasePart") then
                        pcall(function() p.Anchored = false end)
                    end
                end
            end

            -- 10) Final TP player back next to the log
            local liveChar = LocalPlayer.Character
            local liveHrp  = liveChar and liveChar:FindFirstChild("HumanoidRootPart")
            if liveHrp and newLog and newLog.Parent then
                local pp = newLog.PrimaryPart or newLog:FindFirstChild("WoodSection")
                if pp then
                    liveHrp.CFrame = pp.CFrame + Vector3.new(0, 3, 0)
                else
                    liveHrp.CFrame = originCF
                end
            elseif liveHrp then
                liveHrp.CFrame = originCF
            end

            log("Done. Tree at your position.")
            bringTreeActive = false
        end)
    end

    -- LoneCave auto-pickup — faithful port of script4's pattern
    -- (script4 lines 1934-1962, three connections per cut):
    --
    --   1) Humanoid.Died → fire ClientInteracted("Drop tool") on every
    --      Backpack tool so they spawn as MODELS in workspace.PlayerModels.
    --      (Server normally does this on death, but firing manually
    --      guarantees it + lets us track exactly which Models are ours.)
    --
    --   2) workspace.PlayerModels.ChildAdded → watch for owned axe
    --      Models as they spawn. Filter: Owner.Value == LocalPlayer AND
    --      ToolName.Value matches an axe (most are named *Axe*, plus
    --      the BasicHatchet / Rukiryaxe specials).
    --
    --   3) CharacterAdded → wait for respawn, TP back to tree position,
    --      then fire ClientInteracted("Pick up tool") on each tracked
    --      axe Model.
    --
    -- Why a Model and not a Tool: when LT2 drops an axe on death,
    -- the server creates a Model in workspace.PlayerModels containing
    -- a Handle + Owner + ToolName + the visual mesh. We pass THAT
    -- model to ClientInteracted as the target.
    local function setupLoneCaveAutoPickup(treePos)
        local interact = game:GetService("ReplicatedStorage"):FindFirstChild("Interaction")
        local ci = interact and interact:FindFirstChild("ClientInteracted")
        if not ci then return function() end end

        local trackedAxes = {}      -- list of axe Models to pick up
        local diedConn, addConn, charConn

        local function disconnectAll()
            if diedConn then diedConn:Disconnect(); diedConn = nil end
            if addConn  then addConn:Disconnect();  addConn  = nil end
            if charConn then charConn:Disconnect(); charConn = nil end
        end

        -- 1) Pre-arm Died handler: drop all tools so they spawn as Models
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        if hum then
            diedConn = hum.Died:Connect(function()
                log("  ⚠ Player died — dropping tools for re-pickup")
                local bp = LocalPlayer:FindFirstChild("Backpack")
                if bp then
                    for _, tool in ipairs(bp:GetChildren()) do
                        if tool:IsA("Tool") and tool.Name ~= "BlueprintTool" then
                            pcall(function()
                                ci:FireServer(tool, "Drop tool")
                            end)
                        end
                    end
                end
            end)
        end

        -- 2) Pre-arm PlayerModels.ChildAdded to track our dropped axes
        local pm = workspace:FindFirstChild("PlayerModels")
        if pm then
            addConn = pm.ChildAdded:Connect(function(b)
                local owner = b:WaitForChild("Owner", 5)
                local toolName = b:WaitForChild("ToolName", 5)
                if owner and owner.Value == LocalPlayer
                and toolName and toolName.Value
                and (toolName.Value:lower():find("axe")
                     or toolName.Value == "BasicHatchet"
                     or toolName.Value == "Rukiryaxe") then
                    table.insert(trackedAxes, b)
                    log("  · Tracking dropped axe: " .. toolName.Value)
                end
            end)
        end

        -- 3) CharacterAdded → TP back + re-pickup tracked axes
        charConn = LocalPlayer.CharacterAdded:Connect(function(newChar)
            local newHrp = newChar:WaitForChild("HumanoidRootPart", 10)
            if not newHrp then disconnectAll(); return end
            -- Wait for the new character to be fully alive
            local newHum = newChar:WaitForChild("Humanoid", 5)
            task.wait(0.5)

            -- TP back to the tree (script4 fires teleport twice for reliability)
            if treePos then
                pcall(function()
                    newHrp.CFrame = CFrame.new(treePos + Vector3.new(2, 0, 2))
                end)
                task.wait(0.1)
                pcall(function()
                    newHrp.CFrame = CFrame.new(treePos + Vector3.new(2, 0, 2))
                end)
            end
            task.wait(0.3)  -- let position commit

            -- Fire pickup on every tracked axe Model
            local picked = 0
            for _, axe in ipairs(trackedAxes) do
                if axe and axe.Parent then
                    pcall(function() ci:FireServer(axe, "Pick up tool") end)
                    picked = picked + 1
                end
            end
            log("  Auto-pickup: " .. picked .. "/" .. #trackedAxes .. " axes")

            -- ALSO sweep workspace.PlayerModels right now for ANY axe Model
            -- that didn't get tracked (e.g. spawned while addConn was off)
            if pm then
                for _, b in ipairs(pm:GetChildren()) do
                    local own = b:FindFirstChild("Owner")
                    local tn  = b:FindFirstChild("ToolName")
                    if own and own.Value == LocalPlayer and tn and tn.Value
                    and (tn.Value:lower():find("axe")
                         or tn.Value == "BasicHatchet"
                         or tn.Value == "Rukiryaxe")
                    and not table.find(trackedAxes, b) then
                        pcall(function() ci:FireServer(b, "Pick up tool") end)
                        picked = picked + 1
                    end
                end
            end
            disconnectAll()
        end)

        return disconnectAll
    end

    local function bringTreeBatch()
        if bringTreeActive then log("Already running"); return end
        local n = batchCount
        task.spawn(function()
            for i = 1, n do
                if i > 1 then log("─── Tree " .. i .. " / " .. n .. " ───") end

                -- Find the tree FIRST so we can pass its position to the
                -- auto-pickup setup (used to TP back after respawn)
                local treePosForRecovery
                pcall(function()
                    local t = findBestTree(selectedTreeClass)
                    if t and t:FindFirstChild("WoodSection") then
                        treePosForRecovery = t.WoodSection.Position
                    end
                end)

                -- Arm all three watchers (died/drop, model-spawn track,
                -- respawn-pickup) BEFORE the cut starts so we don't miss
                -- the death race.
                local cleanup = setupLoneCaveAutoPickup(treePosForRecovery)

                cutAndBringTree()
                while bringTreeActive do task.wait(0.2) end

                -- If we didn't die during this cut, manually disconnect
                -- the watchers (they'd otherwise hang around).
                cleanup()

                -- Inter-tree settle.
                task.wait(0.6)
            end
            log("✓ Batch complete (" .. n .. " trees)")
        end)
    end

    -- Modern LT2 prices live in ReplicatedStorage.ClientItemInfo.<key>.Price
    -- (verified via Studio: 385 item folders, each with a Price IntValue).
    -- The old Purchasables.Boxes path is gone in 2026.
    -- Direct game:GetService call inside the function so we don't depend
    -- on the `rs` upvalue (which was declared LATER in this scope and
    -- resolved to nil-global at function-define time → silent crash).
    local function getItemPrice(name)
        if not name or name == "" then return nil end
        local cii = game:GetService("ReplicatedStorage"):FindFirstChild("ClientItemInfo")
        if not cii then return nil end
        local entry = cii:FindFirstChild(name)
        if entry then
            local price = entry:FindFirstChild("Price")
            if price then return price.Value end
        end
    end

            abuyPriceLabel.Text = "Price: ? (not in ClientItemInfo)"
            return

        if abuyCount == 1 then
            abuyPriceLabel.Text = string.format("Price: $%d", unit)
        else
            abuyPriceLabel.Text = string.format(
                "Price: $%d × %d = $%d", unit, abuyCount, total)
        end
    end

    -- Find a shop item whose BoxItemName matches (case-insensitive substring).
    -- Skips already-purchased boxes (they have PurchasedBoxItemName instead).
    local function findBuyableItem(name)
        if not name or name == "" then return nil end
        local needle = name:lower()
        local best, bestScore = nil, 0
        for _, m in ipairs(workspace:GetDescendants()) do
            if m:IsA("Model") then
                local box = m:FindFirstChild("BoxItemName")
                if box and box:IsA("StringValue")
                and not m:FindFirstChild("PurchasedBoxItemName") then
                    local hay = box.Value:lower()
                    if hay == needle then
                        return m  -- exact match wins immediately
                    elseif hay:find(needle, 1, true) then
                        -- score by inverse length-diff so shorter matches win
                        local score = 1000 - math.abs(#hay - #needle)
                        if score > bestScore then
                            bestScore = score; best = m
                        end
                    end
                end
            end
        end
        return best
    end

    -- ─────────────────────────────────────────────────────────────
    -- REAL LT2 auto-buy (ported from script4 in scriptsyouwanttoolookover).
    -- The 5-step "Initiate + SetChat" dialog we tried before is for OTHER
    -- NPC interactions (Strange Man toll, etc.). For SHOP items the buy
    -- is a single PlayerChatted:InvokeServer({Cashier,Name,ID},
    -- 'ConfirmPurchase') call, where the cashier ID is dynamic and has
    -- to be gathered at boot by listening to NPCDialog.PromptChat.
    -- ─────────────────────────────────────────────────────────────
    -- Remotes are resolved LAZILY (on every Auto Buy click) with
    -- WaitForChild + 5s timeout. The eager FindFirstChild version
    -- silently failed when the panel loaded before ReplicatedStorage
    -- streamed in NPCDialog/Interaction — which is what was making
    -- "Required LT2 remotes missing" fire even though they exist.
    -- Returns a table of remotes + a list of which ones are missing.
    local rs = game:GetService("ReplicatedStorage")
    local function resolveRemotes()
        local out = {}
        local missing = {}
        local function need(parent, name)
            local r = parent and parent:WaitForChild(name, 5)
            if not r then table.insert(missing, name) end
            return r
        end
        -- Optional remote — patched out in newer LT2 versions. If we don't
        -- find it, we just rely on ClientIsDragging alone, which is what
        -- the existing tree-bring already does successfully.
        local function maybe(parent, name)
            return parent and parent:FindFirstChild(name)
        end
        local npc      = rs:WaitForChild("NPCDialog",   5)
        local interact = rs:WaitForChild("Interaction", 5)
        if not npc      then table.insert(missing, "NPCDialog")   end
        if not interact then table.insert(missing, "Interaction") end
        out.chatted   = need(npc, "PlayerChatted")
        out.setChat   = need(npc, "SetChattingValue")
        out.prompt    = need(npc, "PromptChat")
        out.dragging  = need(interact, "ClientIsDragging")
        out.reqOwn    = maybe(interact, "ClientRequestOwnership")  -- optional (patched out in newer LT2)
        out.ping      = rs:FindFirstChild("TestPing")              -- optional
        out.purchases = rs:FindFirstChild("Purchasables")          -- optional
        return out, missing
    end

    -- Cashier ID lookup. Populated by listening for PromptChat events
    -- after firing SetChat(1)/(0). Each cashier emits its own ID via
    -- the PromptChat event. We retry every time autoBuy is clicked if
    -- the table is empty, in case it failed at boot.
    local cashierIds = {}
    local function gatherCashierIds(prompt, setChat)
        if not prompt or not setChat then return end
        local c = prompt.OnClientEvent:Connect(function(_, data)
            if data and data.Name and cashierIds[data.Name] == nil then
                cashierIds[data.Name] = data.ID
            end
        end)
        pcall(function() setChat:InvokeServer(1) end)
        task.wait(2)
        c:Disconnect()
        pcall(function() setChat:InvokeServer(0) end)
    end
    -- Best-effort gather at boot (asynchronous)
    task.spawn(function()
        local r = resolveRemotes()
        if r.prompt and r.setChat then gatherCashierIds(r.prompt, r.setChat) end
    end)

    -- Map an item name to (store-folder, counter-part) — structure-agnostic.
    -- Walks every descendant of workspace.Stores looking for a Model named
    -- itemName that carries a BoxItemName StringValue. Then walks up to the
    -- shop folder (direct child of Stores) and grabs its Counter (recursive
    -- so nested layouts work too). Replaces script4's old hardcoded-marker
    -- loop that broke when LT2's Stores tree structure changed.
    -- Hardcoded counter positions provided by user. Used as fallback when
    -- walk-up can't find a real Counter Part (LT2 layout has all items in
    -- a shared Stores.ShopItems folder, so the walk-up stops at ShopItems
    -- which has no Counter — these coords map item → correct shop).
    local SHOP_COUNTERS = {
        { name = "WoodRUs",        pos = Vector3.new( 268,    5,    68) },  -- Thom
        { name = "LogicStore",     pos = Vector3.new(4595,    9,  -786) },  -- Links Logic
        { name = "FineArt",        pos = Vector3.new(5238, -164,   740) },  -- Fine Arts
        { name = "FurnitureStore", pos = Vector3.new( 477,    6, -1722) },  -- Fancy Furn
        { name = "CarStore",       pos = Vector3.new( 528,    6, -1460) },  -- Boxed Cars
        { name = "ShackShop",      pos = Vector3.new( 260,   10, -2551) },  -- Bobs Shack
    }
    local function nearestShop(worldPos)
        local best, bestDist
        for _, s in ipairs(SHOP_COUNTERS) do
            local d = (worldPos - s.pos).Magnitude
            if not bestDist or d < bestDist then bestDist = d; best = s end
        end
        return best
    end
    local function itemWorldPos(item)
        local p = item:FindFirstChild("Main") or item.PrimaryPart
        if p and p:IsA("BasePart") then return p.Position end
        for _, d in ipairs(item:GetDescendants()) do
            if d:IsA("BasePart") then return d.Position end
        end
    end

    -- Find a shop display Model whose BoxItemName.Value matches itemName.
    -- Returns (itemsFolder, counterCFrame, shopName) where counterCFrame is
    -- a real CFrame (from a found Counter Part OR from the hardcoded shop
    -- coords above). shopName is a string for the cashier lookup.
    local function getStoreAndCounter(itemName)
        local stores = workspace:FindFirstChild("Stores")
        local searchRoots = {}
        if stores then table.insert(searchRoots, stores) end
        table.insert(searchRoots, workspace)

        local seen = {}
        for _, root in ipairs(searchRoots) do
            for _, d in ipairs(root:GetDescendants()) do
                if not seen[d] then
                    seen[d] = true
                    if d:IsA("Model") then
                        local bin = d:FindFirstChild("BoxItemName")
                        if bin and bin:IsA("StringValue") and bin.Value == itemName then
                            local items = d.Parent

                            -- Strategy A: walk up to find the SHOP FOLDER
                            -- (the direct child of Stores that contains us).
                            -- Then look for Counter inside that shop ONLY —
                            -- never recurse into sibling shops or stop at
                            -- Stores itself (that would name the shop "Stores").
                            local shopFolder
                            if stores then
                                local node = items
                                while node and node.Parent and node.Parent ~= stores do
                                    node = node.Parent
                                end
                                if node and node.Parent == stores then
                                    shopFolder = node
                                end
                            end
                            if shopFolder then
                                local counter = shopFolder:FindFirstChild("Counter", true)
                                             or shopFolder:FindFirstChild("CounterPart", true)
                                             or shopFolder:FindFirstChild("BuyCounter", true)
                                if counter and counter:IsA("BasePart") then
                                    return items, counter.CFrame, shopFolder.Name
                                end
                            end

                            -- Strategy B: fall back to the nearest hardcoded
                            -- shop coord (for the shared Stores.ShopItems layout
                            -- where items have no per-shop ancestor at all).
                            local pos = itemWorldPos(d)
                            if pos then
                                local s = nearestShop(pos)
                                return items, CFrame.new(s.pos), s.name
                            end
                            return nil, nil, "Item '" .. itemName .. "' has no parts to locate"
                        end
                    end
                end
            end
        end
        return nil, nil, "Item with BoxItemName='" .. itemName .. "' not found anywhere"
    end

    -- Find the cashier NPC for a shop. Accepts either:
    --   (a) the items folder — walks up to find the shop folder
    --   (b) a shop name string — looks up workspace.Stores[name]
    -- Then scans the shop's descendants for a humanoid-bearing Model.
    local function getCashier(storeOrName)
        local stores = workspace:FindFirstChild("Stores")
        if not stores then return nil end
        local shop
        if type(storeOrName) == "string" then
            shop = stores:FindFirstChild(storeOrName)
        else
            shop = storeOrName
            while shop and shop.Parent ~= stores do
                shop = shop.Parent
            end
        end
        if not shop then return nil end
        for _, v in ipairs(shop:GetDescendants()) do
            if v:IsA("Model") and v:FindFirstChildOfClass("Humanoid") then
                return v
            end
        end
    end

    -- Money check via leaderstats vs ClientItemInfo price.
    -- (Old Purchasables.Boxes path is gone in modern LT2 — same fix as
    -- getItemPrice above.)
    local function canAfford(_unused, itemName, qty)
        local stats = LocalPlayer:FindFirstChild("leaderstats")
        local money = stats and stats:FindFirstChild("Money")
        if not money then return true end
        local unit = getItemPrice(itemName)
        if not unit then return true end
        return money.Value >= unit * qty
    end

    -- The ACTUAL buy remote — single call with the cashier's dynamic ID
    local function fireBuy(chatted, cashier, id)
        if not chatted or not cashier or not id then return end
        pcall(function()
            chatted:InvokeServer(
                { Character = cashier, Name = tostring(cashier), ID = id },
                "ConfirmPurchase"
            )
        end)
    end

    -- abuyActive / abuyCount declared at the top of this do-block now
    local function autoBuy()
        if abuyActive then log("✗ Already buying — wait"); return end
        if abuyTarget == "" or abuyTarget == "(none found)" then
            log("✗ Pick an item from the dropdown first"); return
        end

        -- Lazy-resolve remotes on every click — if they weren't ready at
        -- boot they almost certainly are by now.
        local R, missing = resolveRemotes()
        if #missing > 0 then
            log("✗ Missing remotes: " .. table.concat(missing, ", "))
            log("  → these LT2 paths weren't found under ReplicatedStorage")
            return
        end

        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then log("✗ No HRP"); return end
        local oldPos = hrp.CFrame

        local store, counterCF, shopName = getStoreAndCounter(abuyTarget)
        if not store or not counterCF then
            log("✗ " .. tostring(shopName or ("Couldn't resolve '" .. abuyTarget .. "'")))
            return
        end
        log("· Shop: '" .. tostring(shopName) .. "' | counter @ "
            .. string.format("(%.0f, %.0f, %.0f)",
                counterCF.X, counterCF.Y, counterCF.Z))
        -- Prefer shopName lookup (more reliable for shared ShopItems layout)
        local cashier = getCashier(shopName) or getCashier(store)
        if not cashier then log("✗ No cashier in shop '" .. tostring(shopName) .. "'"); return end

        -- If we don't have an ID for this cashier yet, try gathering now.
        local cid = cashierIds[tostring(cashier)]
        if not cid then
            log("(no ID for " .. tostring(cashier) .. " — gathering…)")
            gatherCashierIds(R.prompt, R.setChat)
            cid = cashierIds[tostring(cashier)]
            if not cid then
                log("✗ Still no ID — move closer to the shop & retry")
                return
            end
        end
        if not canAfford(R.purchases, abuyTarget, abuyCount) then
            log("✗ Not enough money for " .. abuyCount .. "x " .. abuyTarget)
            return
        end

        log("→ Auto-buying " .. abuyCount .. "x " .. abuyTarget
            .. " from " .. tostring(cashier))
        abuyActive = true

        -- Helper: walk workspace for a fresh Model whose BoxItemName matches.
        -- (Display item is usually one specific instance, but we re-find each
        -- iteration in case the previous one got moved.)
        local function findDisplayItem(name)
            for _, m in ipairs(workspace:GetDescendants()) do
                if m:IsA("Model")
                and not m:FindFirstChild("PurchasedBoxItemName") then
                    local bin = m:FindFirstChild("BoxItemName")
                    if bin and bin:IsA("StringValue") and bin.Value == name then
                        return m
                    end
                end
            end
        end

        -- Cashier position (TP target before firing the buy dialog)
        local function cashierPos()
            local p = cashier:FindFirstChild("HumanoidRootPart")
                   or cashier:FindFirstChild("Torso")
                   or cashier:FindFirstChildWhichIsA("BasePart", true)
            return p and p.Position
        end

        task.spawn(function()
            local bought = 0
            local purchased = {}  -- collected to batch-shuttle home at the end

            -- TP to cashier ONCE up front (was per-item; saves ~300ms × N)
            local cp = cashierPos()
            if cp then
                hrp.CFrame = CFrame.new(cp + Vector3.new(0, 3, 3))
                task.wait(0.05)
            end

            for i = 1, abuyCount do
                -- 1) Find display item (re-find each iter — display swaps)
                local item = findDisplayItem(abuyTarget)
                if not item then log("✗ No display item for '" .. abuyTarget .. "'"); break end

                -- 2) tpOne the item to the counter
                local ok, err = tpOne(item, counterCF + Vector3.new(0, 0.1, 0))
                if not ok then log("✗ Item shuttle: " .. tostring(err)); break end

                -- 3) Hook PlayerModels for the purchased copy
                local detected
                local pm = workspace:FindFirstChild("PlayerModels")
                local conn
                if pm then
                    conn = pm.ChildAdded:Connect(function(b)
                        local owner = b:WaitForChild("Owner", 30)
                        if owner and owner.Value == LocalPlayer then
                            detected = b
                        end
                    end)
                end

                -- 4) Spam ConfirmPurchase EVERY FRAME until detected.
                --    Dropped 100ms wait → task.wait() (~16ms). 2s timeout
                --    (was 5s) since purchases register in <500ms normally.
                if R.ping then pcall(function() R.ping:InvokeServer() end) end
                local t0 = tick()
                while not detected and tick() - t0 < 2 do
                    fireBuy(R.chatted, cashier, cid)
                    task.wait()
                end
                if conn then conn:Disconnect() end

                if not detected then
                    log("✗ Buy " .. i .. " timed out — no purchase detected")
                    break
                end

                -- 5) Quick MoveTo to clear the counter so the next item can
                --    land. NOT a full tpOne (~400ms saved per iter). The
                --    final batched shuttle at the end positions everything
                --    properly at oldPos.
                pcall(function() detected:MoveTo(oldPos.Position + Vector3.new(i*3, 0, 0)) end)
                if dragRemote then
                    task.spawn(function()
                        for _ = 1, 8 do
                            if not detected or not detected.Parent then break end
                            pcall(function() dragRemote:FireServer(detected) end)
                            task.wait()
                        end
                    end)
                end
                table.insert(purchased, detected)
                bought = bought + 1
                log("✓ Bought " .. i .. "/" .. abuyCount)
            end

            -- 6) Batched return shuttle — tpOne each purchased item to
            --    oldPos. Spreads them in a small ring so they don't stack.
            for j, m in ipairs(purchased) do
                if m and m.Parent then
                    local angle = (j / math.max(1, #purchased)) * math.pi * 2
                    local target = oldPos
                                 + Vector3.new(math.cos(angle) * 3, 0, math.sin(angle) * 3)
                    tpOne(m, target)
                end
            end

            -- Send player back to origin
            local c = LocalPlayer.Character
            local h = c and c:FindFirstChild("HumanoidRootPart")
            if h then h.CFrame = oldPos end
            log("✓ Auto Buy done — " .. bought .. "/" .. abuyCount .. " purchased")
            abuyActive = false
        end)
    end

    -- Scan = walk workspace, collect every distinct unpurchased BoxItemName
    -- (Layout orders: dropdown=1, pick toggle=2, Scan=3, Buy=4, qty label=5, slider=6, hint=7)
    makeButtonRow(abuyCard, "Refresh dropdown with shop items", "Scan Items", function()
        local seen, list = {}, {}
        for _, m in ipairs(workspace:GetDescendants()) do
            if m:IsA("Model") then
                local box = m:FindFirstChild("BoxItemName")
                if box and box:IsA("StringValue")
                and not m:FindFirstChild("PurchasedBoxItemName") then
                    if not seen[box.Value] then
                        seen[box.Value] = true
                        table.insert(list, box.Value)
                    end
                end
            end
        end
        table.sort(list)
        if #list == 0 then
            log("No shop items found — go near a shop first")
            availableBuys = {"(none found)"}
        else
            log("Found " .. #list .. " unique items — pick one in the dropdown")
            availableBuys = list
            if not table.find(list, abuyTarget) then
                abuyTarget = list[1]
                abuyBtn.Text = " " .. abuyTarget .. "         ▾"
            end
        end
        rebuildAbuyList()
        updatePriceLabel()
    end, 4)

    makeButtonRow(abuyCard, "Buy the item picked in the dropdown", "Auto Buy", autoBuy, 5)

    -- Public hook for other sections: buys `qty` of `name` synchronously,
    -- returns when autoBuy's async task finishes.
    buyOneByName = function(name, qty)
        if abuyActive then return false, "busy" end
        abuyTarget = name
        abuyCount  = qty or 1
        autoBuy()
        local deadline = tick() + 30
        while abuyActive and tick() < deadline do task.wait(0.1) end
        return true
    end

        abuyCount = v

    -- Never Lose Axe — pre-emptive stash. LT2's death-drop loop iterates
    -- Character + Backpack and parents each Tool to workspace. Tools
    -- parented to LocalPlayer (the Player object itself, not Backpack /
    -- Character) are invisible to that loop — they stay safe across death
    -- and respawn. We watch HP; when it drops, we stash. On respawn we
    -- restore. We do NOT void-fall (LT2's anti-cheat snaps the teleport
    -- and you still die normally on the way).
    local nlaOn = false
    local nlaConns = {}
    local nlaStashed = false

    -- Move every axe-like Tool to LocalPlayer (out of Backpack/Character)
    local function stashToPlayer()
        local n = 0
        local bp = LocalPlayer:FindFirstChild("Backpack")
        if bp then
            for _, t in ipairs(bp:GetChildren()) do
                if t:IsA("Tool") and t:FindFirstChild("ToolName") then
                    pcall(function() t.Parent = LocalPlayer end)
                    n = n + 1
                end
            end
        end
        local char = LocalPlayer.Character
        if char then
            for _, t in ipairs(char:GetChildren()) do
                if t:IsA("Tool") and t:FindFirstChild("ToolName") then
                    pcall(function() t.Parent = LocalPlayer end)
                    n = n + 1
                end
            end
        end
        if n > 0 then nlaStashed = true end
        return n
    end

    -- After respawn, walk every Tool currently parented to LocalPlayer
    -- back to Backpack so the user can equip them again.
    local function restoreToBackpack()
        local bp = LocalPlayer:WaitForChild("Backpack", 5)
        if not bp then return end
        local n = 0
        for _, t in ipairs(LocalPlayer:GetChildren()) do
            if t:IsA("Tool") and t:FindFirstChild("ToolName") then
                pcall(function() t.Parent = bp end)
                n = n + 1
            end
        end
        nlaStashed = false
        if n > 0 then log("Never Lose Axe: restored " .. n .. " axe(s) to Backpack") end
    end

    -- Hook the Humanoid: stash when HP drops or any damage fires.
    -- We force-unequip first because equipped Tools are in Character (not
    -- Backpack) and Roblox's death pipeline auto-drops them to workspace
    -- BEFORE Died fires — too late for us. Unequipping first pulls them
    -- back to Backpack where our stash sweep catches them.
    -- HP threshold set HIGH (any damage at all) since LT2 has one-shot
    -- damage sources (lava, fall, sword crit).
    local function hookHumanoid(char)
        if not char or not nlaOn then return end
        local hum = char:FindFirstChildOfClass("Humanoid")
                 or char:WaitForChild("Humanoid", 5)
        if not hum then return end
        local maxHp = hum.MaxHealth
        local c1 = hum.HealthChanged:Connect(function(hp)
            -- Trigger on ANY damage (any drop from MaxHealth)
            if not nlaOn or hp <= 0 then return end
            if hp < maxHp and not nlaStashed then
                pcall(function() hum:UnequipTools() end)
                stashToPlayer()
            end
        end)
        table.insert(nlaConns, c1); bind(c1)
        local c2 = hum.Died:Connect(function()
            if not nlaOn then return end
            pcall(function() hum:UnequipTools() end)
            stashToPlayer()
        end)
        table.insert(nlaConns, c2); bind(c2)
    end

    -- Orphan watcher: if an axe lands in workspace owned by us (the

                if not own or own.Value == LocalPlayer then
                    pcall(function() obj.Parent = LocalPlayer end)
                end
            end
        end)

    end

    -- Dupe Axe (BARK HUB normalinvdupe — script9 line 2851).
    -- This is the dupe that bark used for years, the gold-standard LT2
    -- exploit. The death-tick / save-load desync attempts are patched
    -- (server enforces CurrentlySavingOrLoading lock), but this one
    -- exploits a DIFFERENT mechanic: silent re-spawn via Character
    -- mutilation.
    --
    -- Algorithm:
    --   1) Save HRP CFrame as oldCF
    --   2) Humanoid:UnequipTools() — pulls equipped axe to Backpack
    --   3) HRP.CFrame = oldCF (snap-pin)
    --   4) Destroy every Character child EXCEPT Humanoid
    --      → no Died event fires (because Humanoid still exists)
    --      → but Character has no Head/Torso/etc so it can't function
    --   5) For each tool in Backpack:
    --        - HRP.CFrame = oldCF (re-pin in case it moved)
    --        - Fire ClientInteracted("Drop tool", tool, oldCF)
    --      → every axe drops to workspace AT oldCF
    --   6) Wait for CharacterAdded — server respawns you BUT loads
    --      inventory from save data instead of "tools you had"
    --      (because no Died fired, the death-drop loop didn't run)
    --   7) WaitForChild("HumanoidRootPart") + WaitForChild("Backpack.Tool")
    --   8) HRP.CFrame = oldCF — TP new char back so they walk over the
    --      dropped axes and pick them up
    --
    -- Net: original axes spawn in Backpack from save restore + dropped
    -- copies still on the floor = duplicates.
    --
    -- The slider lets you repeat the dupe N times in one click (each
    -- pass roughly doubles your axe count).
    local dupeRepeat = 1
    -- Slider 1-5: number of dupe passes (each pass ~doubles axe count)
    makeSliderRow(hpCard, 1, 1, 5, function(v) dupeRepeat = math.floor(v) end, 7)
    -- Dupe Axe (Save + Stash + Load pattern — BotSneaky Store/Restore).
    -- This is the classic LT2 axe dupe that actually works:
    --   1) Save current state to slot N → save data now has all axes
    --   2) Wait for the CurrentlySavingOrLoading lock to clear
    --   3) "Stash" — reparent every Backpack Tool to LocalPlayer (root).
    --      Server only looks at Backpack/Character for inventory state,
    --      so it can't see the stashed tools (they're "outside" the
    --      inventory).
    --   4) Load slot N → server reads save data and CREATES new Tool
    --      instances in Backpack from the save (it doesn't pre-clear
    --      Backpack, but Backpack is empty now anyway after stash)
    --   5) Wait for load to complete
    --   6) "Unstash" — reparent stashed tools back to Backpack
    --   7) Net: stashed (1 copy) + restored from save (1 copy) = 2 copies
    -- Each pass doubles your tool count. 5 passes = 32x your starting set.
    --
    -- This works because RequestLoad doesn't:
    --   - Walk LocalPlayer's direct children
    --   - Validate "did we have these tools before"
    --   - Block if the player has a Tool outside Backpack
    makeButtonRow(hpCard, "Save+Stash+Load dupe (classic Store/Restore)", "Dupe Axe", function()
        local rs2 = game:GetService("ReplicatedStorage")
        local lsr = rs2:FindFirstChild("LoadSaveRequests")
        local saveR = lsr and lsr:FindFirstChild("RequestSave")
        local loadR = lsr and lsr:FindFirstChild("RequestLoad")
        if not (saveR and loadR) then
            log("✗ Need LoadSaveRequests.RequestSave + RequestLoad")
            return
        end
        local currentSlotObj = LocalPlayer:FindFirstChild("CurrentSaveSlot")
        local SLOT = currentSlotObj and currentSlotObj.Value or 1
        if SLOT == -1 then SLOT = 1 end  -- never loaded? default to slot 1
        local lockObj = LocalPlayer:FindFirstChild("CurrentlySavingOrLoading")

        local function waitLock(deadline)
            if not lockObj then task.wait(0.5); return end
            local t0 = tick()
            while lockObj.Value and tick() - t0 < deadline do
                task.wait(0.05)
            end
        end

        local function countAxes()
            local n = 0
            local bp = LocalPlayer:FindFirstChild("Backpack")
            if bp then
                for _, t in ipairs(bp:GetChildren()) do
                    if t:IsA("Tool") and t.Name ~= "BlueprintTool" then n = n + 1 end
                end
            end
            local char = LocalPlayer.Character
            if char then
                for _, t in ipairs(char:GetChildren()) do
                    if t:IsA("Tool") and t.Name ~= "BlueprintTool" then n = n + 1 end
                end
            end
            return n
        end

        local function runOnce(passIdx)
            local before = countAxes()
            if before == 0 then
                log("✗ Pass " .. passIdx .. ": no axes in inventory — pick one up first")
                return false
            end

            local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            if hum then pcall(function() hum:UnequipTools() end) end
            task.wait()

            -- 1) Save current state (so save has the axes)
            waitLock(3)
            local saveOk = false
            pcall(function()
                saveOk = saveR:InvokeServer(SLOT, LocalPlayer)
            end)
            if not saveOk then
                log("✗ Pass " .. passIdx .. ": RequestSave returned false (slot " .. SLOT .. " busy?)")
                return false
            end
            waitLock(4)
            log("  Pass " .. passIdx .. ": saved slot " .. SLOT .. " (had " .. before .. " axes)")

            -- 2) STASH — move every Backpack Tool to LocalPlayer root
            local stashed = {}
            local bp = LocalPlayer:FindFirstChild("Backpack")
            if bp then
                for _, t in ipairs(bp:GetChildren()) do
                    if t:IsA("Tool") and t.Name ~= "BlueprintTool" then
                        pcall(function() t.Parent = LocalPlayer end)
                        table.insert(stashed, t)
                    end
                end
            end
            log("  Pass " .. passIdx .. ": stashed " .. #stashed .. " axes")

            -- 3) LOAD — server restores Backpack from save data.
            --    Pre-flight ClientMayLoad first (mirrors the official
            --    LoadSaveClient flow at line 354) — server rejects
            --    RequestLoad without it. Hook CharacterAdded BEFORE
            --    firing so we don't miss the respawn race.
            waitLock(3)
            local mayLoad = lsr:FindFirstChild("ClientMayLoad")
            if mayLoad then
                local ok, allowed, err = pcall(function()
                    return mayLoad:InvokeServer(LocalPlayer)
                end)
                if not ok or not allowed then
                    log("✗ Pass " .. passIdx .. ": ClientMayLoad rejected: " .. tostring(err))
                    for _, t in ipairs(stashed) do
                        if t and t.Parent == LocalPlayer then
                            pcall(function() t.Parent = bp end)
                        end
                    end
                    return false
                end
            end

            -- Pre-arm the respawn wait
            local respawnTriggered = false
            local respawnConn = LocalPlayer.CharacterAdded:Connect(function()
                respawnTriggered = true
            end)

            local loadOk = false
            pcall(function()
                loadOk = loadR:InvokeServer(SLOT, LocalPlayer, nil)
            end)
            if not loadOk then
                respawnConn:Disconnect()
                log("✗ Pass " .. passIdx .. ": RequestLoad returned false — unstashing")
                for _, t in ipairs(stashed) do
                    if t and t.Parent == LocalPlayer then
                        pcall(function() t.Parent = bp end)
                    end
                end
                return false
            end

            -- Wait for CharacterAdded (load completes by respawning you
            -- at the saved spot). Falls back to lock-clear if respawn
            -- doesn't fire (load may not respawn if you didn't move).
            local respDeadline = tick() + 10
            while not respawnTriggered and tick() < respDeadline
                  and lockObj and lockObj.Value do
                task.wait(0.1)
            end
            respawnConn:Disconnect()
            waitLock(8)

            -- Wait for the new Backpack to actually exist + populate
            local newBp = LocalPlayer:WaitForChild("Backpack", 8)
            if newBp then
                -- Give server time to insert restored Tools
                local popDeadline = tick() + 5
                while tick() < popDeadline do
                    local hasRestored = false
                    for _, t in ipairs(newBp:GetChildren()) do
                        if t:IsA("Tool") and t.Name ~= "BlueprintTool" then
                            hasRestored = true; break
                        end
                    end
                    if hasRestored then break end
                    task.wait(0.1)
                end
            end

            -- 4) UNSTASH — server may have purged LocalPlayer Tools too
            --    (some LT2 versions clear ALL player Tools, not just
            --    Backpack, before applying save). Diagnose where each
            --    stashed Tool actually went, then try to revive it from
            --    wherever it landed.
            local restoredCount = 0
            local diag = {}
            if newBp then
                for _, t in ipairs(stashed) do
                    if t then
                        local p = t.Parent
                        local where
                        if p == nil then where = "nil"
                        elseif p == LocalPlayer then where = "Player"
                        elseif p == newBp then where = "Backpack"
                        elseif p == LocalPlayer.Character then where = "Character"
                        else where = tostring(p) end
                        table.insert(diag, t.Name .. "(" .. where .. ")")
                        -- Try to reparent IF it still has an identity
                        if p ~= newBp then
                            pcall(function() t.Parent = newBp end)
                        end
                        if t.Parent == newBp then
                            restoredCount = restoredCount + 1
                        end
                    else
                        table.insert(diag, "(destroyed)")
                    end
                end
            end
            log("  Pass " .. passIdx .. ": unstashed " .. restoredCount .. "/"
                .. #stashed .. " | stashed Tools went to: " .. table.concat(diag, ", "))

            -- If 0/N, the server is clearing LocalPlayer too. As a last
            -- resort, fire ClientInteracted("Pick up tool") on any Tool
            -- left in workspace near our position (server may have
            -- dropped the stashed Tools to the floor instead of purging).
            if restoredCount == 0 and #stashed > 0 then
                local interact = rs2:FindFirstChild("Interaction")
                local ci = interact and interact:FindFirstChild("ClientInteracted")
                local char = LocalPlayer.Character
                local hrp  = char and char:FindFirstChild("HumanoidRootPart")
                if ci and hrp then
                    local pickedUp = 0
                    for _, obj in ipairs(workspace:GetChildren()) do
                        if obj:IsA("Tool") and obj:FindFirstChild("Handle") then
                            if (obj.Handle.Position - hrp.Position).Magnitude < 50 then
                                pcall(function()
                                    ci:FireServer(obj, "Pick up tool")
                                end)
                                pickedUp = pickedUp + 1
                            end
                        end
                    end
                    if pickedUp > 0 then
                        log("    (fallback: picked up " .. pickedUp
                            .. " nearby workspace Tools)")
                    end
                end
            end

            local after = countAxes()

    -- ── Wood Finder ─────────────────────────────────────────────
    -- Scans every TreeRegion for trees of a given TreeClass. Useful
    -- for rare classes (LoneCave only spawns in the Maze Cave region,
    -- Spooky/SpookyNeon only at specific times). Logs total count +
    -- closest distance + TPs you next to the closest one.
    -- Returns # found OR triggers server-hop if 0. When called from the
    -- pending-find handler (after a hop), it'll auto-hop again if THIS
    -- server also has no matches — guaranteed to keep hopping until it
    -- finds one or you click Stop.
    local function findAndTPToTree(treeClass, autoHopOnMiss)
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then log("✗ No HRP"); return 0 end
        local found = {}
        for _, region in ipairs(workspace:GetChildren()) do
            if region.Name:sub(1, 11) == "TreeRegion" or region.Name == "TreeRegion" then
                for _, tree in ipairs(region:GetChildren()) do
                    local tc = tree:FindFirstChild("TreeClass")
                    local own = tree:FindFirstChild("Owner")
                    if tc and tc.Value == treeClass
                    and (not own or not own.Value or own.Value == LocalPlayer) then
                        local ws = tree:FindFirstChild("WoodSection")
                        if ws then
                            table.insert(found, {tree = tree, ws = ws,
                                dist = (ws.Position - hrp.Position).Magnitude})
                        end
                    end
                end
            end
        end
        if #found > 0 then
            table.sort(found, function(a, b) return a.dist < b.dist end)
            local closest = found[1]
            log(string.format("✓ Found %d %s tree(s) — TPing to closest (%.0f studs)",
                #found, treeClass, closest.dist))
            pcall(function()
                hrp.CFrame = CFrame.new(closest.ws.Position + Vector3.new(5, 5, 0))
            end)
            return #found
        end

        log("✗ No " .. treeClass .. " trees in this server")
        if autoHopOnMiss then
            log("→ Server-hopping to find " .. treeClass .. "…")
            -- Queue the same find on the next server so it auto-continues
            local extra = string.format(
                "_G.LT2_PendingFind = %q\n", treeClass)
            local ok, err = serverHop(extra)
            if not ok then
                log("✗ Server-hop failed: " .. tostring(err))
            end
        end
        return 0
    end

    -- Public hook for the pending-find handler at script top (runs after
    -- a hop completes and the script re-injects)
    _G.FindTreeAndHop = function(treeClass)
        findAndTPToTree(treeClass, true)
    end

    bpListLayout.Padding = UDim.new(0, 2); bpListLayout.Parent = bpList

    local function rebuildBpList()
        for _, c in ipairs(bpList:GetChildren()) do
            if c:IsA("TextButton") then c:Destroy() end
        end
        for _, nm in ipairs(availablePlayers) do
            local o = Instance.new("TextButton")
            o.Size = UDim2.new(1, -8, 0, 20); o.AutoButtonColor = false
            o.BackgroundTransparency = 1
            o.Text = "  " .. nm; o.TextXAlignment = Enum.TextXAlignment.Left
            o.Font = Fonts.Mono; o.TextSize = 11; o.ZIndex = 51; o.Parent = bpList
            themeify(o, "TextColor3", "Text"); corner(o, 3)
            o.MouseEnter:Connect(function()
                o.BackgroundTransparency = 0; o.BackgroundColor3 = Theme.SurfaceAlt
            end)
            o.MouseLeave:Connect(function() o.BackgroundTransparency = 1 end)
            o.MouseButton1Click:Connect(function()
                selectedTargetName = nm
                bpBtn.Text = " " .. nm .. "         ▾"
                bpList.Visible = false
            end)
        end
    end
    bpBtn.MouseButton1Click:Connect(function()
        if bpList.Visible then bpList.Visible = false; return end
        local abs, sz = bpBtn.AbsolutePosition, bpBtn.AbsoluteSize
        bpList.Position = UDim2.fromOffset(abs.X, abs.Y + sz.Y + 4)
        bpList.Visible = true
    end)
    bind(UserInputService.InputBegan:Connect(function(i)
        if not bpList.Visible then return end
        if i.UserInputType ~= Enum.UserInputType.MouseButton1
        and i.UserInputType ~= Enum.UserInputType.Touch then return end
        local mx, my = i.Position.X, i.Position.Y
        local function within(o)
            local p, s = o.AbsolutePosition, o.AbsoluteSize
            return mx >= p.X and mx <= p.X + s.X and my >= p.Y and my <= p.Y + s.Y
        end
        if not within(bpList) and not within(bpBtn) then bpList.Visible = false end
    end))

    -- Scan = repopulate the dropdown with current players (minus self)
    makeButtonRow(bpCard, "Refresh server player list", "Scan Players", function()
        availablePlayers = {}
        for _, plr in ipairs(game:GetService("Players"):GetPlayers()) do
            if plr ~= LocalPlayer then
                table.insert(availablePlayers, plr.Name)
            end
        end
        table.sort(availablePlayers)
        if #availablePlayers == 0 then
            log("No other players in server")
            availablePlayers = {"(empty)"}
        else
            log("Found " .. #availablePlayers .. " players")
            if not table.find(availablePlayers, selectedTargetName) then
                selectedTargetName = availablePlayers[1]
                bpBtn.Text = " " .. selectedTargetName .. "         ▾"
            end
        end
        rebuildBpList()
    end, 2)

    -- Go To Player — teleports YOU to them. Always works because we
    -- own our own HRP.
    makeButtonRow(bpCard, "TP yourself to target", "Go To Player", function()
        if selectedTargetName == "" or selectedTargetName == "(empty)" then
            log("✗ Pick a player first"); return
        end
        local plr = game:GetService("Players"):FindFirstChild(selectedTargetName)
        if not plr or not plr.Character then log("✗ " .. selectedTargetName .. " not in game"); return end
        local theirHrp = plr.Character:FindFirstChild("HumanoidRootPart")
        local myChar = LocalPlayer.Character
        local myHrp  = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not theirHrp or not myHrp then log("✗ Missing HRP"); return end
        myHrp.CFrame = theirHrp.CFrame + Vector3.new(0, 5, 0)
        log("✓ Went to " .. selectedTargetName)
    end, 3)

    -- Bring (Car) — click-pick a car, then click Bring. Flow:
    --   1) Pick Car: arms a click handler. Next mouse click on a car
    --      stores it as bpSelectedCar (walks up to the top-most Model
    --      containing a seat).
    --   2) Bring: saves the car's current CFrame, then uses the PROVEN
    --      tpOne shuttle to slide the car so its passenger seat lands on
    --      the target. Same tpOne pipeline Bring Tree / Auto Sell use.
    --   3) Waits up to 8s for the target's Humanoid.SeatPart to be one
    --      of the car's passenger seats.
    --   4) tpOne shuttles the car (with passenger) back to the saved CFrame.
    -- You do NOT need to sit in the car — script handles ownership via
    -- ClientIsDragging spam inside tpOne.
    local bpSelectedCar = nil
    local bpPickingCar  = false

    local bpCarLabel = Instance.new("TextLabel")
    bpCarLabel.LayoutOrder = 3; bpCarLabel.Size = UDim2.new(1, 0, 0, 16)
    bpCarLabel.BackgroundTransparency = 1
    bpCarLabel.Text = "Car: (none)"
    bpCarLabel.Font = Fonts.Mono; bpCarLabel.TextSize = 11
    bpCarLabel.TextXAlignment = Enum.TextXAlignment.Left
    bpCarLabel.Parent = bpCard
    themeify(bpCarLabel, "TextColor3", "Text")

    -- Walk up the parent chain to the top-most Model containing the part
    local function findCarModel(part)
        local cur = part and part.Parent
        local topModel = nil
        while cur and cur ~= workspace do
            if cur:IsA("Model") then topModel = cur end
            cur = cur.Parent
        end
        return topModel
    end

    -- All seats in the car (any VehicleSeat / Seat under the car Model)
    local function getCarSeats(car)
        local seats = {}
        if not car then return seats end
        for _, d in ipairs(car:GetDescendants()) do
            if d:IsA("VehicleSeat") or d:IsA("Seat") then
                table.insert(seats, d)
            end
        end
        return seats
    end

    bind(UserInputService.InputBegan:Connect(function(input, processed)
        if processed or not bpPickingCar then return end
        if input.UserInputType ~= Enum.UserInputType.MouseButton1 then return end
        local mouse = LocalPlayer:GetMouse()
        if not mouse or not mouse.Target then return end
        local model = findCarModel(mouse.Target)
        if not model then log("✗ No Model under that part"); return end
        -- Verify it has at least one seat (so it's actually a vehicle)
        local seats = getCarSeats(model)
        if #seats == 0 then log("✗ '" .. model.Name .. "' has no seats"); return end
        bpSelectedCar = model
        bpCarLabel.Text = "Car: " .. model.Name .. " (" .. #seats .. " seats)"
        log("✓ Car picked: " .. model.Name)
        bpPickingCar = false
    end))

    makeButtonRow(bpCard, "Click a car part in workspace to select it", "Pick Car", function()
        bpPickingCar = true
        log("→ Click on the car you want to use")
    end, 4)

    local bringActive = false
    makeButtonRow(bpCard, "Shuttle picked car to target + back via tpOne", "Bring", function()
        if bringActive then log("✗ Already shuttling — wait"); return end
        if selectedTargetName == "" or selectedTargetName == "(empty)" then
            log("✗ Pick a player first"); return
        end
        if not bpSelectedCar or not bpSelectedCar.Parent then
            log("✗ Pick a car first (Pick Car button)"); return
        end
        local plr = game:GetService("Players"):FindFirstChild(selectedTargetName)
        if not plr or not plr.Character then log("✗ Not in game"); return end
        local theirHum = plr.Character:FindFirstChildOfClass("Humanoid")
        local theirHrp = plr.Character:FindFirstChild("HumanoidRootPart")
        if not theirHum or not theirHrp then log("✗ Target has no humanoid"); return end

        local seats = getCarSeats(bpSelectedCar)
        if #seats == 0 then log("✗ Picked car has no seats anymore"); return end

        local savedCF = bpSelectedCar:GetPivot()
        log("Car: '" .. bpSelectedCar.Name .. "' (" .. #seats .. " seats) → " .. selectedTargetName)
        bringActive = true

        task.spawn(function()
            -- Phase 1: compute the car pivot that lands a seat on the
            -- target, then use tpOne to shuttle the car there.
            --
            -- Math: we want seat.CFrame to end at the target's CFrame.
            -- PivotTo moves the whole model rigidly:
            --   newSeatCF = newPivot * (oldPivot:Inverse() * oldSeatCF)
            -- Solve for newPivot:
            --   newPivot  = targetSeatCF * (oldPivot:Inverse() * oldSeatCF):Inverse()
            local seat         = seats[1]
            local oldPivot     = bpSelectedCar:GetPivot()
            local seatOffset   = oldPivot:Inverse() * seat.CFrame
            local targetSeatCF = theirHrp.CFrame + Vector3.new(0, 0.5, 0)
            local newPivot     = targetSeatCF * seatOffset:Inverse()

            log("→ tpOne shuttling car to " .. selectedTargetName .. "…")
            local ok, err = tpOne(bpSelectedCar, newPivot)
            if not ok then log("(car shuttle: " .. tostring(err) .. ")") end

            -- Phase 2: wait up to 8s for them to enter ANY seat in the car
            log("Waiting for " .. selectedTargetName .. " to enter the car…")
            local t0 = tick()
            local seated = false
            while tick() - t0 < 8 do
                local sp = theirHum.SeatPart
                if sp then
                    for _, s in ipairs(seats) do
                        if sp == s then seated = true; break end
                    end
                end
                if seated then break end
                task.wait(0.1)
            end

            if not seated then
                log("✗ " .. selectedTargetName .. " never got in — aborted")
                bringActive = false
                return
            end

            log("✓ " .. selectedTargetName .. " is in — tpOne'ing car back")
            task.wait(0.2)

            -- Phase 3: tpOne the car (with passenger) back to saved CFrame
            local ok2, err2 = tpOne(bpSelectedCar, savedCF)
            if not ok2 then log("(return shuttle: " .. tostring(err2) .. ")") end
            log("✓ Brought " .. selectedTargetName)
            bringActive = false
        end)
    end, 5)

    local bpHint = Instance.new("TextLabel")
    bpHint.LayoutOrder = 6; bpHint.Size = UDim2.new(1, 0, 0, 86)
    bpHint.BackgroundTransparency = 1
    bpHint.Text = "1) Scan Players + pick a target\n2) Pick Car + click any part of the car in workspace\n3) Bring — tpOne shuttles the car to the target,\n   waits up to 8s for them to seat, then tpOne shuttles\n   the car (with passenger) back to its origin."
    bpHint.Font = Fonts.Regular; bpHint.TextSize = 11
    bpHint.TextXAlignment = Enum.TextXAlignment.Left
    bpHint.TextWrapped = true; bpHint.Parent = bpCard
    themeify(bpHint, "TextColor3", "TextDim")

    end  -- close Bring Player local-scope

    -- ── Status log section ──────────────────────────────────────
    makeSection(p, "Status Log", 17)
    local logCard = makeCard(p, 18)
    statusLabel = Instance.new("TextLabel")
    statusLabel.Size = UDim2.new(1, 0, 0, 80)
    statusLabel.BackgroundTransparency = 1
    statusLabel.Text = "(waiting for actions…)"
    statusLabel.Font = Fonts.Mono; statusLabel.TextSize = 11
    statusLabel.TextXAlignment = Enum.TextXAlignment.Left
    statusLabel.TextYAlignment = Enum.TextYAlignment.Top
    statusLabel.TextWrapped = true
    statusLabel.Parent = logCard
    themeify(statusLabel, "TextColor3", "Text")

    -- Boot diagnostics
    task.spawn(function()
        task.wait(0.5)
        if dragRemote then
            log("✓ Boot: dragRemote ready")
        else
            log("⏳ Boot: waiting for dragRemote…")
        end
    end)

    _G.PanelClearSelection = clearAll
    controls.SelMode = { toggle = selToggle }
end

-- Full Reset helper

local function fullReset()
    for _,t in ipairs(Tweaks) do
        t.value=t.default
        if t.enabled and controls[t.key] and controls[t.key].toggle then controls[t.key].toggle.set(false) end
        if controls[t.key] and controls[t.key].slider then controls[t.key].slider.setVisual(t.default) end
        t:restore(Humanoid)
    end
    if controls.Fly        and controls.Fly.toggle.get()        then controls.Fly.toggle.set(false) end
    if controls.InfJump    and controls.InfJump.toggle.get()    then controls.InfJump.toggle.set(false) end
    if controls.Fullbright and controls.Fullbright.toggle.get() then controls.Fullbright.toggle.set(false) end
    if controls.Noclip     and controls.Noclip.toggle.get()     then controls.Noclip.toggle.set(false) end
    if controls.SelMode    and controls.SelMode.toggle.get()    then controls.SelMode.toggle.set(false) end
    if _G.PanelClearSelection then _G.PanelClearSelection() end
    if refreshStatusDot then refreshStatusDot() end
end
local function teardown()
    for _,t in ipairs(Tweaks) do if t.enabled then t.enabled=false; t:restore(Humanoid) end end
    flySetEnabled(false); fullbrightSet(false); InfJump.enabled=false; noclipSet(false)
    if _G.PanelClearSelection then _G.PanelClearSelection() end
    unbindAll(); ScreenGui:Destroy()
end

            resizing=false end
    end))
